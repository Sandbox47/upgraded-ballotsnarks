\section{Algebraic Functions and Algebraic Circuits}
An algebraic circuit models an algebraic function $r$ that maps from a set of inputs to a set of outputs.
%An algebraic circuit represents a relation between a set of input signals and a set of output signals from a fixed field.
\begin{definition}[Algebraic Function]
    \label{def:algebraicFunction}
    Let $(\mathbb{F}, +, \cdot)$ be a finite field.
    Let $m, n\in\mathbb{N}$ where $m$ is the number of input signals and $n$ is the number of output signals.
    Then, an algebraic function is defined as:
    \begin{align*}
        r:\mathbb{F}^m\to \mathbb{F}^n;\:(S_1^{in},\dots, S_{m}^{in})\mapsto (S_1^{out},\dots, S_n^{out})
    \end{align*}
    Here, the function needs to be expressible using the basic operations $+$ and $\cdot$ of $\mathbb{F}$.
\end{definition}

In order to see how algebraic functions can be used to indicate ballot validity, we consider our example from \cref{chap:votingschemes}.
Here, among other things, a voter needs to show that the ballot $b=(b_0,b_1,b_2)$ they chose is part of the choice space $C=\{(0,0,0),(1,0,0), (0,1,0), (0,0,1)\}$.
We can express this membership of the choice space by the function $r_{3\text{-}vot}$ defined in \cref{example:threeVot}.
\begin{example}[Choice space membership for $3\text{-}vot$]
    \label{example:threeVot}
    Let $(\mathbb{F},+,\cdot)$ be a finite field of sufficiently large characteristic \cref{footnote:choiceSpaceSize} with $0,1\in\mathbb{F}$.
    Then, $r_{3\text{-}vot}$ is defined as:
    \begin{align*}
        r_{3\text{-}vot}:\mathbb{F}^3\to\mathbb{F}^4;\quad(S_1^{in},S_2^{in}, S_3^{in})\mapsto (S_1^{out}, S_2^{out}, S_3^{out}, S_4^{out})
    \end{align*}
    Here, we can compute $(S_1^{out}, S_2^{out}, S_3^{out}, S_4^{out})=r_{3\text{-}vot}(S_1^{in},S_2^{in}, S_3^{in})$ as in \cref{alg:threeVot}.
    Note, that $r_{3\text{-}vot}(b_0,b_1,b_2)=(0,0,0,0)\Leftrightarrow (b_0,b_1,b_2)\in C$.

    $r_{3\text{-}vot}$ can be represented by the circuit $\mathfrak{C}_{3\text{-}vot}$ in \cref{fig:circuit:exampleVoting}.
\end{example}

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Function $r_{3\text{-}vot}$ used to check whether a ballot $b=(b_0,b_1,b_2)=(S_1^{in}, S_2^{in}, S_3^{in})$ is in the choice space $C=\{(0,0,0),(1,0,0),(0,1,0),(0,0,1)\}$. 
    $r_{3\text{-}vot}$ returns $(0,0,0,0)$ if this is the case and something else otherwise.}
    \label{alg:threeVot}
    \begin{algorithmic}
    \Procedure{$r_{3\text{-}vot}$}{$S_1^{in}, S_2^{in}, S_3^{in}$}
        \State $S_1^{out}\gets (1-S_1^{in})\cdot S_1^{in}$
        \State $S_2^{out}\gets (1-S_2^{in})\cdot S_2^{in}$
        \State $S_3^{out}\gets (1-S_3^{in})\cdot S_3^{in}$ \Comment Determine whether the individual ballots are bits.
        \State $tmp\gets S_1^{in}+S_2^{in}+S_3^{in}$
        \State $S_4^{out}\gets (1-tmp)\cdot tmp$ \Comment Determine whether the sum of all ballots is a bit.
        \State \Return $(S_1^{out}, S_2^{out}, S_3^{out}, S_4^{out})$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

\subsection{Satisfiability}

Now, we can assign arbitrary input and output values to the circuit $\mathfrak{C}$ for the function $r$.
If we then compute the algebraic function $r$ on the input values, this either yields the chosen output values or something else.
If the computation results in the same output values as the ones chosen before, we say that $\mathfrak{C}$ is satisfied.
% Furthermore, we can fix some of the input and output values of the circuit.
% Then it is either possible to assign values to the rest of the signals such that the circuit for $r$ holds or it is not possible to do so.
We can formalize these ideas with the concept circuit satisfiability.
%

    Let $r:\mathbb{F}^m\to \mathbb{F}^n$ be an algebraic function.
    Let $S=(S_1^{in}, \dots, S_{m}^{in}, S_1^{out},\dots, S_{n}^{out})\in\mathbb{F}^{m+n}$ be an assignment to $r$.

    We say that $S$ satisfies the circuit $\mathfrak{C}$ for $r$, iff $r(S_1^{in},\dots, S_{m}^{in})=(S_1^{out},\dots, S_{n}^{out})$.
    We call $\mathfrak{C}$ satisfiable if there exists an assignment $S$ such that $S$ satisfies $\mathfrak{C}$.
    % We define the set of all assignments $S$ that satisfy $r$ as the language of $r$: $L_r\{S\in\mathbb{F}^{m+n}|S \text{ satisfies } r\}$.

    % \end{align*}

    %     S'^{in}_j=I_{j}^{out} \:\text{ if }\: I_{j}^{out}\neq \bot
    % \end{align*}
\end{definition}

\subsection{Asserts}

In general, an assignment $(S_1^{in}, S_2^{in}, S_3^{in}, S_1^{out}, S_2^{out}, S_3^{out}, S_4^{out}):=(b_0, b_1, b_2, 0, 0, 0, 0)$ satisfies $\mathfrak{C}_{3\text{-}vot}$ if and only if $b=(b_0, b_1, b_2)\in C$.
Therefore, to assert that ballots are part of the choice space, we assign the constant $0$ to $S_1^{out}, S_2^{out}, S_3^{out}$ and $S_4^{out}$.
This creates a new circuit $\mathfrak{C}_{Assert\text{-}3\text{-}vot}$ which operates only on the three input signals $S_1^{in}, S_2^{in}$ and $S_3^{in}$ and does not have any outputs.
$\mathfrak{C}_{Assert\text{-}3\text{-}vot}$ is illustrated in \cref{fig:circuit:exampleAssertThreeVot}.
An assignment $(S_1^{in}, S_2^{in}, S_3^{in})$ satisfies $\mathfrak{C}_{Assert\text{-}3\text{-}vot}$ if and only if $b=(b_0,b_1,b_2)=(S_1^{in}, S_2^{in}, S_3^{in})\in C$.

\subsection{Modularity}

In many cases, we need the same functionality at multiple places in an algebraic function $r$ and its corresponding circuit $\mathfrak{C}$.
For instance, in $r_{3\text{-}vot}$ and $r_{Assert\text{-}3\text{-}vot}$, this applies to the computation needed to check whether a signal is a bit:
$(1-S)\cdot S$ and $(1-S)\cdot S\mbeq 0$ respectively is used for all individual votes $b_0,b_1,b_2$ in a ballot $b$ and for the sum of all votes $b_0+b_1+b_2$.
We can capture the check that a signal is a bit inside another algebraic function $r_{Is\text{-}Bit}$:
\begin{align*}
    r_{Is\text{-}Bit}:\mathbb{F}\to\mathbb{F}, S_1^{in}\to S_1^{out}:=(1-S_1^{in})\cdot S_1^{in}=\begin{cases}
        0,&\quad\text{if } S_1^{in}\in\{0,1\}\\
        x\in\mathbb{F}\setminus\{0\},&\quad\text{otherwise}
    \end{cases}
\end{align*}

Similarly, we can capture the assertion that signals are bits in an algebraic function $r_{Assert\text{-}Bit}$.
Here, we can already reuse our definition of $r_{Is\text{-}Bit}$:
\begin{align*}
    r_{Assert\text{-}Bit}:\mathbb{F}\to\emptyset, \text{ with}%r_{assert\text{-}Bit}(S_1^{in}):=\left(r_{Is\text{-}Bit}(S_1^{in})\mbeq 0\right)
\end{align*}
\begin{algorithmic}
    \Procedure{$r_{Assert\text{-}Bit}$}{$S_1^{in}$}
        \State $0\mbeq r_{Is\text{-}Bit}(S_1^{in})$
    \EndProcedure
\end{algorithmic}

\section{Rank-1 Constraint Systems}
Fundamentally, \glspl{r1cs} are systems of equations of a specific format over a set of variables.

\begin{definition}[Rank-1 Constraint System]
    \label{def:r1cs}
    Let $(\mathbb{F},+,\cdot )$ be a field, let $n,k\in\mathbb{N}$.
    Let $a_j^i,b_j^i,c_j^i\in\mathbb{F}$ be constants from $\mathbb{F}$ for every index $i\in\{0,\dots, n\}$ and $j\in\{0,\dots, k\}$.
    Then, a \gls{r1cs} $R1CS$ is defined as the following set of $k$ equations over the variables $S_1,\dots, S_n\in\mathbb{F}$:
    \begin{align*}
        \left(a_0^1+\sum\limits_{j=1}^n a_j^1\cdot S_j\right)\cdot \left(b_0^1+\sum\limits_{j=1}^n b_j^1\cdot S_j\right) \:&=\: c_0^1+\sum\limits_{j=1}^n c_j^1\cdot S_j\\
        &\vdotswithin{=}\\
        \left(a_0^k+\sum\limits_{j=1}^n a_j^k\cdot S_j\right)\cdot \left(b_0^k+\sum\limits_{j=1}^n b_j^k\cdot S_j\right) \:&=\: c_0^k+\sum\limits_{j=1}^n c_j^k\cdot S_j\\
    \end{align*}
    Hence, $n$ is the number of variables involved in the \gls{r1cs} and $k$ is the number of equations or constraints of the system.
\end{definition}

Note, that we can represent any algebraic function $r:\mathbb{F}^m\to \mathbb{F}^n$ and therefore also any algebraic circuit $\mathfrak{C}$ as a \gls{r1cs} $R1CS_r$.
We require that $r$ and $\mathfrak{C}$ can be expressed using just the basic field operations $+$ and $\cdot $ of $\mathbb{F}$.

\begin{definition}[\gls{r1cs} Satisfiability]
    \label{def:r1csSat}
    Let $(\mathbb{F},+,\cdot )$ be a field, let $n,k\in\mathbb{N}$.
    Let $a_j^i,b_j^i,c_j^i\in\mathbb{F}$ be constants from $\mathbb{F}$ for every index $i\in\{0,\dots, n\}$ and $j\in\{0,\dots, k\}$.
    Let $R1CS$ be the corresponding \gls{r1cs} as in \cref{def:r1cs}.
    Let $S=(S_1,\dots, S_n)\in\mathbb{F}^{n}$ be an assignment to $R1CS$.
    Then, we say that $S$ satisfies $R1CS$, if and only if all $k$ equations from \cref{def:r1cs} are fulfilled.    

    We call $R1CS$ satisfiable, if there is an assignment $S$ that satisfies $R$.
\end{definition}

As we have shown before, the circuit $\mathfrak{C}_{3\text{-}vot}$ for $r_{3\text{-}vot}$ is satisfied by the assignment $S=(0,0,1,0,0,0,0)$ but not by the assignment $S'=(1,1,0,0,0,0,0)$.
Therefore, the \gls{r1cs} $R1CS_{r_{3\text{-}vot}}$ should also be satisfied by $S$ but not by $S'$.
We can check this by inserting the values for the variables $S_1,\dots, S_7$ into the equations of $R1CS_{r_{3\text{-}vot}}$:
\begin{itemize}
    \item Assignment $(0,0,1,0,0,0,0)$:
    \begin{align*}
        \left(1-0\right)\cdot \left(0\right) = \left(0\right)\\
        \left(1-0\right)\cdot \left(0\right) = \left(0\right)\\
        \left(1-1\right)\cdot \left(1\right) = \left(0\right)\\
        \left(1-0-0-1\right)\cdot \left(0+0+1\right) = \left(0\right)\\
    \end{align*}
    \item Assignment $(1,1,0,0,0,0,0)$:
    \begin{align*}
        \left(1-1\right)\cdot \left(1\right) = \left(0\right)\\
        \left(1-1\right)\cdot \left(1\right) = \left(0\right)\\
        \left(1-0\right)\cdot \left(0\right) = \left(0\right)\\
        \left(1-1-1-0\right)\cdot \left(0+1+1\right) = -2 \neq \left(0\right)\\
    \end{align*}
\end{itemize}

In general, a ballot $b=(b_0,b_1,b_2)$ is in the choice space $C$ if and only if $(b_0,b_1,b_2,0,0,0,0)$ satisfies $R1CS_{r_{3\text{-}vot}}$.
Therefore, we can use $R1CS_{r_{3\text{-}vot}}$ to show membership of the choice space similarly to how we used $\mathfrak{C}_{3\text{-}vot}$.

\section{Quadratic Arithmetic Programs}
To use \glspl{r1cs} in \glspl{gpzkp}, we need to transform them into another different representation, namely \glspl{qap}.
For every \gls{r1cs} $R1CS$, there is a corresponding \gls{qap} $QAP(R1CS)$.

\begin{definition}[Quadratic Arithmetic Program]
    \label{def:qap}
    Let $(\mathbb{F},+,\cdot )$ be a field, let $n,k\in\mathbb{N}$.
    Let $a_j^i,b_j^i,c_j^i\in\mathbb{F}$ be constants from $\mathbb{F}$ for every index $i\in\{0,\dots, n\}$ and $j\in\{0,\dots, k\}$.
    Let $R1CS$ be the corresponding \gls{r1cs} as in \cref{def:r1cs}.
    Let $m_1,\dots, m_k$ be arbitrary, invertible and distinct elements of $\mathbb{F}$.
    Then, the \gls{qap} corresponding to the \gls{r1cs} $R1CS$ is defined as follows:
    \begin{align*}
        QAP(R1CS)=\left\{T\right\}\cup \bigcup_{j=0}^{n}\left\{\left(A_j, B_j, C_j\right)\right\} %\text{ with} T\in\mathbb{F}[x]\:\forall j\in\{0,\dots, n\} A_j,B_j,C_j\in\mathbb{F}[x]
    \end{align*}
    Here, $T, A_j, B_j, C_j\in\mathbb{F}[x]$ are polynomials over $\mathbb{F}$ with:
    \begin{itemize}
        \item $T(x):=\prod\limits_{l=1}^k(x-m_l)$ is a polynomial of degree $k$.
        \item $A_j, B_j, C_j$ are polynomials of degree $k-1$ defined by:
        \begin{align*}
            \forall j\in\{0,\dots, n\}\forall i\in\{1,\dots, k\}: \:A_j(m_i)=a_j^{i},\:B_j(m_i)=b_j^{i},\: C_j(m_i)=c_j^{i}
        \end{align*}
    \end{itemize}
\end{definition}

We can construct $QAP(R1CS)$ from $R1CS$ by executing the following steps:
\begin{enumerate}
    \item Choose $k$ distinct, invertible elements $m_1,\dots, m_k\in\mathbb{F}$, one for each constraint in $R1CS$. \footnote{
        \label{footnote:qapRootsOfUnity}
        In practice, we typically use the $k$-th roots of unity $\{m_1,\dots, m_k\}=\{m\in\mathbb{F}| m^k=1\}$ for efficiency reasons. \cite{dunkelman_families_2022}
    }
    \item Compute the target polynomial $T$ as in \cref{def:qap}.
    \item Use Lagrange's method of polynomial interpolation as described in Section 3.3.4 of \cite{least_authority_moonmath_2024} to compute the polynomials $A_j, B_j$ and $C_j$ as in \cref{def:qap}.
    Here, we need to use the set $S_{A_j}:=\{(m_1,a_j^1),\dots, (m_k, a_j^k)\}$ as the (evaluation point, value)-pairs for the construction of $A_j$. 
    Analogously, we use $S_{B_j}:=\{(m_1,b_j^1),\dots, (m_k, b_j^k)\}$ and $S_{C_j}:=\{(m_1,c_j^1),\dots, (m_k, c_j^k)\}$ for the construction of $B_j$ and $C_j$.
\end{enumerate}

To construct the corresponding $QAP(R1CS_{r_{3\text{-}vot}})$, we first need to choose the elements $m_1,m_2,m_3,m_4\in\mathbb{F}$.

\begin{definition}[\gls{qap} Satisfiability]
    \label{def:qapSat}
    Let $(\mathbb{F},+,\cdot )$ be a field, let $n,k\in\mathbb{N}$.
    Let $a_j^i,b_j^i,c_j^i\in\mathbb{F}$ be constants from $\mathbb{F}$ for every index $i\in\{0,\dots, n\}$ and $j\in\{0,\dots, k\}$.
    Let $R1CS$ be the corresponding \gls{r1cs} as in \cref{def:r1cs} and 
    \begin{align*}
        QAP(R1CS)=\left\{T\right\}\cup \bigcup_{j=0}^{n}\left\{\left(A_j, B_j, C_j\right)\right\}
    \end{align*}
    be the corresponding \gls{qap}.\\
    Let $S=(S_1,\dots, S_n)\in\mathbb{F}^n$ be an assignment to $R1CS$.
    We compute a polynomial $P_S$ as follows:
    \begin{align*}
        P_S:=\left(A_0+\sum_{j=1}^{n}S_j\cdot A_j\right)\cdot\left(B_0+\sum_{j=1}^{n}S_j\cdot B_j\right)-\left(C_0+\sum_{j=1}^{n}S_j\cdot C_j\right)
    \end{align*}
    Then, we say that $S$ satisfies $QAP(R1CS)$, if and only if $P_S\mod T = 0$.

    We call $QAP(R1CS)$ satisfiable, if there is an assignment $S$ that satisfies $QAP(R1CS)$.
\end{definition}

In general, the satisfiability definition of circuits implies the satisfiability definition of \glspl{r1cs} and \glspl{qap} in the sense that for every assignment that satisfies the circuit, there is an extended assignment, where values are assigned to the intermediate signals as well, that satisfies the \gls{r1cs} and the \gls{qap} as presented in \cref{theorem:satisfiabilityEquality}.
\begin{theorem}[Satisfiability Definitions Relation]
    \label{theorem:satisfiabilityEquality}
    Let $(\mathbb{F}, +, \cdot )$ be a finite field.
    Let $r:\mathbb{F}^m\to \mathbb{F}^n$ be an algebraic function and $\mathfrak{C}$ be the corresponding algebraic circuit.
    Let $R1CS_r$ be the corresponding \gls{r1cs} and let $QAP(R1CS_r)$ be the corresponding \gls{qap} with $s$ additional intermediate signals.
    Without loss of generality, assume that the order of assignments to $\mathfrak{C}$ is preserved by $R1CS_r$ and $QAP(R1CS_r)$ and the additional intermediate signals are added as a suffix.\footnote{
        \label{footnote:assignmentOrderDoesNotMatter}
        Intuitively, the order of the assignments to a \gls{r1cs} and corresponding \gls{qap} does not matter since we can just swap where the different signals from an assignment are being used in the two representations.
    }

    Let $S=(S_1^{in},\dots, S_m^{in}, S_1^{out}, \dots, S_n^{out})$ be an assignment that satisfies $\mathfrak{C}$.
    Then, there are signals $S_1\in\mathbb{F},\dots, S_s\in\mathbb{F}$ such that $S'=(S_1^{in},\dots, S_m^{in}, S_1^{out}, \dots, S_n^{out}, S_1,\dots, S_s)$ satisfies $R1CS_r$ and $QAP(R1CS_r)$.
\end{theorem}

We will not prove this theorem here formally.
However, intuitively, the intermediate signals $S_1,\dots, S_s$ can be computed from $S_1^{in},\dots, S_m^{in}$ in the same way as in $\mathfrak{C}$.

\section{Instance and Witness}
\label{section:proofSchemes:instanceAndWitness}
In practical applications such as proving ballot validity, we need to be able to conduct the proof without revealing the secret ballot. 
This is captured by the zero knowledge characteristic of ballot validity proofs in the voting schemes introduced in \cref{chap:votingschemes}.
More generally, we need to be able to hide a part of the assignment to a circuit $\mathfrak{C}$ for a relation $r$ and still have to be able to provide the proof of correct circuit execution.

For this purpose, we split an assignment to the circuit $\mathfrak{C}$ for $r$ into an instance and a witness part. 
As outlined in \cref{chap:votingschemes}, the instance is public, and the witness is private and proving the relation between them is the goal of the \gls{zps} (\cref{def:zpsForR}) in our voting schemes.

We can formalize this concept as a Decomposition Relation.

\begin{definition}[Decomposition Relation]
    Let $(\mathbb{F}, +, \cdot)$ be a finite field.
    Let $r:\mathbb{F}^m\to\mathbb{F}^n$ be an algebraic function.
    Then, $R1CS_r$ is the corresponding \gls{r1cs} and $QAP(R1CS_r)$ is the corresponding \gls{qap}.

    Assignments to $r$ have $m+n$ entries, to account for the input and output signals.
    For $R1CS_r$ and $QAP(R1CS_r)$, assignments have more entries if intermediate signals are required to conform to the \gls{r1cs} and \gls{qap} format.
    The set of all assignments to any of the representations $r$, $R1CS_r$ and $QAP(R1CS_r)$ is therefore $\mathcal{A}=\mathbb{F}^{m+n+s}$, where $s\in\mathbb{N}$ is the number of intermediate signals required.
    
    To decompose an assignment $S\in\mathcal{A}$ into an assignment $(I,W)\in\mathcal{I}\times\mathcal{W}$, we employ a bijective function $d_p$ based on a permutation $p:\{0,\dots, m+n+s\}\to\{0,\dots, n_I + n_W\}$ with $m+n+s=n_I+n_W$ and $p(0):=0$. \footnote{\label{footnote:decompRelationPermutation}
        To understand how $p$ helps to map between instance-witness decomposition and normal assignments, consider the following:
        Let $S=(S_1,\dots, S_{m+n+s})$ be an assignment, and let $(I,W)=((I_1,\dots, I_{n_I}), (W_1,\dots, W_{n_W}))$ be the corresponding decomposed assignment, where the signals $S$ have each been assigned to instance $I$ or witness $W$.
        If $I_j$ corresponds to $S_i$, then $p(i):=j$ and consequently $p^{-1}(j)=i$.
        Similarly, if $W_j$ corresponds to $S_i$, then $p(i)=n_I+j$ and consequently $p^{-1}(n_I+j)=i$.
        (We need to add $n_I$ to the index of the witness signal to account for all $n_I$ instance signals before the witness signals.)

        The definition $p(0):=0$ is purely to simplify notation in \glspl{g16snark} later on. 
        For this section, defining $p:\{1,\dots, m+n+s\}\to\{1,\dots, n_I + n_W\}$ would be sufficient as well.
    }
    \begin{align*}
        &d_p:\mathcal{A}\to\mathcal{I}\times\mathcal{W},\\
        &\quad d_p(S)=d_p((S_1,\dots, S_{m+n+s})):=(I,W)=((I_1,\dots,I_{n_I}),(W_1,\dots, W_{n_W}))\\
        &\text{where}\\
        &\forall j\in\{1,\dots,{n_I}\}:\:I_j=S_{p^{-1}(j)}\\
        &\forall j\in\{1,\dots,{n_W}\}:\: W_j=S_{p^{-1}({n_I}+j)}
    \end{align*}
    How $p$ and $d_p$ are used to decompose assignments is also visualized in \cref{fig:decompositionPerm}.
    % The decomposition $(I,W)$ of an assignment $S=(S_1,\dots, S_{m+n+s})$ is then defined as
    % \begin{align*}
    %     (I,W)&\phantom{:}=((I_1,\dots, I_u), (W_1,\dots, W_v))\\
    %     &:= ((S_{p^{-1}(1)},\dots, S_{p^{-1}(u)}))
    % \end{align*}
    % We need this, to map an assignment to the instance and witness parts to a full assignment of the circuit, \gls{r1cs} or \gls{qap} for $r$.
    % Furthermore, we use a permutation $p:\mathcal{I}\times\mathcal{W}\to \mathcal{A}$ to map an assignment to the instance and witness parts to a full assignment of the circuit for $r$.

    The decomposition relation $DR(rep, d_p)$ where $rep\in\{r, R1CS_r, QAP(R1CS_r)\}$ is then a relation defined as
    \begin{align*}
        &DR(rep, d_p): \mathcal{I}\times\mathcal{W}\to\{0,1\}, \\
        &\quad DR(rep,d_p)(I, W)=DR(rep, d_p)((I_1,\dots, I_{n_I}), (W_1,\dots, W_{n_W}))\\
        &\phantom{\quad DR(rep,d_p)(I, W)}=\begin{cases}
            %1, \quad \text{if } p(I, W) \text{ satisfies } rep\\
            1, \quad \text{if } d_p^{-1}((I_1,\cdots, I_{n_I}), (W_1,\dots, W_{n_W})) \text{ satisfies } rep\\
            0, \quad \text{otherwise}
        \end{cases}
    \end{align*}
    This fulfills the relation definition from \cref{def:relation}.
    %We say that an assignment $(I,W)=((I_1,\dots, I_u), (W_1,\dots, W_v))\in\mathbb{F}^{n_I}\times\mathbb{F}^{n_W}$ satisfies $DR(rep)$, if and only if $S=p(I,W)$ satisfies $rep$.
    %$DR(rep)$ is called satisfiable, if there is an assignment $(I,W)\in\mathbb{F}^{n_I}\times\mathbb{F}^{n_W}$ such that $(I,W)$ satisfies $DR(rep)$.
\end{definition}

Once again, we consider $r_{3\text{-}vot}$ as an example. 
To show ballot validity with the circuit $\mathfrak{C}_{3\text{-}vot}$ for $r_{3\text{-}vot}$, we want to keep the ballot represented by the inputs $S_1^{in}, S_2^{in}$ and $S_3^{in}$ private.
On the other hand, the output signals $S_1^{out}, S_2^{out}, S_3^{out}$ and $S_4^{out}$ need to be $0$ in case we have a valid ballot as we observed earlier.
Therefore, these signals can be public.
To compute the corresponding decomposition relation, we choose $\mathcal{I}=\mathbb{F}_{13}^{4}$ and $\mathcal{W}=\mathbb{F}_{13}^3$.
Additionally, we choose permutation $p$ and consequently mapping $d_p$ as follows:
\begin{align*}
    &p:\{0,\dots, 7\}\to\{0,\dots, 7\},0\mapsto0,1\mapsto5,2\mapsto6,3\mapsto7,4\mapsto1,5\mapsto2,6\mapsto3,7\mapsto4
    &\\
    &d_p:\mathbb{F}_{13}^7\to\mathbb{F}_{13}^4\times \mathbb{F}_{13}^3, d_p(S_1,S_2,S_3,S_4,S_5,S_6,S_7) = ((S_4,S_5,S_6, S_7),(S_1,S_2,S_3))
\end{align*}
Then, $DR(r_{3\text{-}vot}, d_p)$ is the corresponding decomposition relation.

Let $DR(QAP(R1CS_{r_{3\text{-}vot}}), d_p)$ be defined analogously to $DR(r_{3\text{-}vot}, d_p)$. \footnote{\label{footnote:instanceWitnessIntermediateSignals}
    Note, that in our case, there are no intermediate signals that need to be assigned to instance or witness. 
    However, in many circuits there are such signals. 
    In that case, we typically determine whether an intermediate signal is instance or witness depending on the instance and witness properties of the rest of the signals.
    If an intermediate signal is only dependent on instance signals, then it is typically also part of the instance.
    Otherwise, the intermediate signal is usually assigned to the witness.

To prove that a ballot $b=(b_0,b_1,b_2)$ is in the choice space $C$ without revealing it, a voter can create the assignment $S=d_p^{-1}((0,0,0,0),(b_0,b_1,b_2))=(b_0,b_1,b_2,0,0,0,0)$ to $QAP(R1CS_{r_{3\text{-}vot}})$ and compute the corresponding polynomial $P_S$.
If $P_S$ is divisible by the target polynomial $T$ of $QAP(R1CS_{r_{3\text{-}vot}})$, we know that the assignment $S$ satisfies $QAP(R1CS_{r_{3\text{-}vot}})$.
Therefore, $DR(QAP(R1CS_{r_{3\text{-}vot}}), d_p)(I,W)=1$ and the ballot $b$ is valid.
However, we do not learn what the ballot $b$ is, which is what the zero-knowledge property of our voting scheme definition captures.
Such constructive proofs of circuit execution are the basis for \glspl{g16snark}.

\section{Groth16 SNARKs}
\label{section:proofSchemes:groth16}
\glspl{snark} are a special form of the \glspl{zkp} we introduced in \cref{chap:votingschemes}.

\begin{definition}[Succinct Non-Interactive Argument of Knowledge for Relation $R$]
    Let $R$ be a binary polynomial time decidable relation relating instances $I\in\mathcal{I}$ and witnesses $W\in\mathcal{W}$.
    Let $ZPS(R)=(Setup, Prove, Verify)$ be a \gls{zps} for $R$.
    $ZPS(R)$ is a \gls{snark}, if it fulfills the following properties:\footnote{\label{footnote:NonInteractiveness}
        Note that our definition of \glspl{zkp} is already non-interactive, meaning that a prover sends only the proof $\pi$ to the verifier.
        There is no further interaction between the two parties.
    }
    \begin{itemize}
        \item Succinctness: The size of the proof $\pi=Prove(CRS, I, W)$ and the time needed to compute $Verify(CRS, I, \pi)$ are both polynomially bound \footnote{
            \label{footnote:securityParameterSuccinctness}
            Here, "polynomially bound"\ means polynomially bound with regard to the security parameter $\lambda$ \cite{groth_size_2016}.
            A security parameter $\lambda$ is used to define the security level of cryptographic constructions.
        }\cite{groth_size_2016}.
        \item Argument: The soundness property defined in \cref{def:zpsForR} only has to hold if $Prove$ is polynomially bound\cref{footnote:securityParameterSuccinctness} \cite{least_authority_moonmath_2024}.
        Note that this slightly weakens our previous definition.
    \end{itemize}
\end{definition}

One widely used \gls{snark} is the Groth16 SNARK. 
Our definition largely parallels the one in Section 8.2 of \cite{least_authority_moonmath_2024} and the one in Appendix B in \cite{huber_zk-snarks_2025}.

%\newpage
\begin{definition}[Groth16 SNARK]
    Let $\mathbb{F}_q$ be a field of prime order $q$.
    Let $r:\mathbb{F}_q^m\to\mathbb{F}_q^n$ be an algebraic function.
    Let 
    \begin{align*}
        QAP(R1CS)=\left\{T\right\}\cup \bigcup_{j=0}^{m+n+s}\left\{\left(A_j, B_j, C_j\right)\right\}
    \end{align*}
    be a \gls{qap} corresponding to $r$.
    We denote the number of constraints in $QAP(R1CS_r)$ by $k:=deg(T)$.
    Let $DR(QAP(R1CS_r), d_p):\mathcal{I}\times\mathcal{W}\to\{0,1\}$ be a corresponding decomposition relation where $d_p:\mathcal{A}\to\mathcal{I}\times\mathcal{W}$, $\mathcal{I}=\mathbb{F}_q^{n_I}, \mathcal{W}=\mathbb{F}_q^{n_W}$ where $n_I + n_W=m+n+s$ and $\mathcal{A}=\mathbb{F}_q^{m+n+s}$.

    Let $(\mathbb{G}_1, \odot_{\mathbb{G}_1}), (\mathbb{G}_2, \odot_{\mathbb{G}_2})$ and $(\mathbb{G}_T, \odot_{\mathbb{G}_T})$ be groups of prime order $q$.
    Let $e:\mathbb{G}_1\times\mathbb{G}_2\to\mathbb{G}_T$ be an efficiently computable, non-degenerate, bilinear pairing as defined in \cref{def:bilinearPairing}.
    Let $g_1\in\mathbb{G}_1, g_2\in\mathbb{G}_2$ be generators of the respective groups and $e(g_1, g_2)$ be a generator of $\mathbb{G}_T$.

    Then, the \gls{g16snark} for the relation $DR(QAP(R1CS_r), d_p)$ based on the parameters $params:=(\mathbb{G}_1, \mathbb{G}_2, e(\cdot,\cdot),g_1,g_2,\mathbb{F}_q)$ is the \gls{snark}
    \begin{align*}
        GSNARK(DR(QAP(R1CS_r), d_p))=(Setup, Prove, Verify)
    \end{align*}
    where the procedures $Setup$, $Prove$ and $Verify$ are defined in Algorithms \ref{alg:setupG16}, \ref{alg:proveG16} and \ref{alg:verifyG16} respectively.
    Here, the different components used in the algorithms are colored to see where they are being used, similarly as in \cite{huber_zk-snarks_2025}.
\end{definition}

The zero-knowledge property of \glspl{zps} can be defined formally for \glspl{g16snark}.
In their work, Groth \cite{groth_size_2016} introduced an additional algorithm $Sim(I, (\alpha,\beta,\gamma,\delta,\tau))$ which produces a forged proof $\pi'$ for $I$.
A detailed definition of $Sim$ is presented in \cite{groth_size_2016} and also in Appendix B of \cite{huber_zk-snarks_2025}.
Here, $Sim$ uses an instance $I$ and the simulation trapdoor $(\alpha, \beta, \delta, \gamma, \tau)$ from $Setup$ (\cref{alg:setupG16}) but does not receive a witness $W$.
Based on this algorithm, the zero-knowledge property of \glspl{g16snark} can be defined as follows \cite{groth_size_2016}:
Let $CRS$ be a common reference string and let $I\in\mathcal{I}$ and $W\in\mathcal{W}$. % such that $I$ and $W$ are related, i.e. $DR(QAP(R1CS_r), d_p)=1$.
Then, any adversary cannot distinguish the proofs $\pi:=Prove(CRS, I, W)$ and $\pi':=Sim(CRS, I, (\alpha, \beta, \gamma, \delta, \tau))$.
In other words, the proof $\pi$ does not reveal anything about the witness $W$ since $W$ was not involved in the creation of $\pi'$.

Proofs of Knowledge Soundness, Completeness and Succinctness as well as a proof of the Zero-Knowledge property of \glspl{g16snark} are presented in \cite{groth_size_2016}.

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Procedure to set up the \gls{zps} $GSNARK(DR(QAP(R1CS_r), d_p))$ based on $params:=(\mathbb{G}_1, \mathbb{G}_2, e(\cdot,\cdot),g_1,g_2,\mathbb{F}_q)$.
    Here, $r:\mathbb{F}_q^m\to\mathbb{F}_q^n$ and $DR(QAP(R1CS_r), d_p):\mathcal{I}\times\mathcal{W}\to\{0,1\}$, where $d_p:\mathcal{A}\to\mathcal{I}\times\mathcal{W}$, $\mathcal{I}=\mathbb{F}_q^{n_I}, \mathcal{W}=\mathbb{F}_q^{n_W}$ with $n_I + n_W=m+n+s$ and $\mathcal{A}=\mathbb{F}_q^{m+n+s}$.
    Additionally, $QAP(R1CS)=\left\{T\right\}\cup \bigcup_{j=0}^{n}\left\{\left(A_j, B_j, C_j\right)\right\}$ and $k:=deg(T)$ is the number of constraints in the \gls{qap}.
    }
    \label{alg:setupG16}
    \begin{algorithmic}
    \Procedure{$Setup$}{$\vphantom{bla}$}
        \State $\alpha, \beta, \gamma, \delta, \tau\probgets \mathbb{F}_q$
        \State $st\gets (\alpha,\beta,\gamma,\delta,\tau)$ \Comment Create simulation trapdoor from uniformly random field elements.
        \State $CRS_{\mathbb{G}_1}\gets \left\{\textcolor{red}{g_1^\alpha},\textcolor{blue}{g_1^\beta},\textcolor{lime}{g_1^\delta},\left(\textcolor{teal}{g_1^{\tau^j}}\right)_{j=0}^{k-1},\left(\textcolor{cyan}{g_1^{\frac{\beta A_{p^{-1}(j)}(\tau)+\alpha B_{p^{-1}(j)}(\tau)+C_{p^{-1}(j)}(\tau)}{\gamma}}}\right)_{j=0}^{n_I},\right.$
        \Statex $\phantom{CRS_{\mathbb{G}_1}\gets \quad\quad} \left. \left(\textcolor{magenta}{g_1^{\frac{\beta A_{p^{-1}(j+n_I)}(\tau)+\alpha B_{p^{-1}(j+n_I)}(\tau)+C_{p^{-1}(j+n_I)}(\tau)}{\delta}}}\right)_{j=1}^{n_W},\left(\textcolor{brown}{g_1^{\frac{\tau^jT(\tau)}{\delta}}}\right)_{j=0}^{k-2}\right\}$
        \State $CRS_{\mathbb{G}_2}\gets \left\{\textcolor{olive}{g_2^\beta}, \textcolor{green}{g_2^\gamma},\textcolor{violet}{g_2^\delta},\left(\textcolor{pink}{g_2^{\tau^j}}\right)_{j=0}^{k-1}\right\}$
        \State $CRS\gets (CRS_{\mathbb{G}_1},CRS_{\mathbb{G}_2})$ \Comment Compute Common Reference String
        \State \Return $CRS$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Procedure to compute a proof for a \gls{g16snark} using a $CRS$ as computed by \cref{alg:setupG16}, instance $I$ and witness $W$.
    Here, the \gls{g16snark} $GSNARK(DR(QAP(R1CS_r), d_p))$ is based on $params:=(\mathbb{G}_1, \mathbb{G}_2, e(\cdot,\cdot),g_1,g_2,\mathbb{F}_q)$.
    Moreover, $r:\mathbb{F}_q^m\to\mathbb{F}_q^n$ and $DR(QAP(R1CS_r), d_p):\mathcal{I}\times\mathcal{W}\to\{0,1\}$, where $d_p:\mathcal{A}\to\mathcal{I}\times\mathcal{W}$, $\mathcal{I}=\mathbb{F}_q^{n_I}, \mathcal{W}=\mathbb{F}_q^{n_W}$ with $n_I + n_W=m+n+s$ and $\mathcal{A}=\mathbb{F}_q^{m+n+s}$.
    Additionally, $QAP(R1CS)=\left\{T\right\}\cup \bigcup_{j=0}^{m+n+s}\left\{\left(A_j, B_j, C_j\right)\right\}$ and $k:=deg(T)$ is the number of constraints in the \gls{qap}.
    }
    \label{alg:proveG16}
    \begin{algorithmic}
    \Procedure{$Prove$}{$CRS, I, W$}
        \State $(I_1,\dots, I_{n_I}:=I);\:(W_1,\dots, W_{n_W}):=W$
        \State $(S_1,\dots, S_{n_I+n_W}):=S=d_p^{-1}(I,W)$
        \State Compute the polynomial $P_S$ corresponding to assignment $S$ as in \cref{def:qapSat}.
        \State $H\gets \frac{P_S}{T}$ with $H(x)=\sum\limits_{l=0}^{k-2}H_{l}x^{l}$ \Comment $H_{l}$ is the $l$-th coefficient of $H$.
        \Statex 
        \State Compute $\textcolor{NavyBlue}{g_1^S}:=\textcolor{NavyBlue}{g_1^{\frac{H(\tau)T(\tau)}{\delta}}}=\prod\limits_{j=0}^{k-2}\left(\textcolor{brown}{g_1^{\frac{\tau^jT(\tau)}{\delta}}}\right)^{H_j}$
        \State $\textcolor{yellow}{g_1^{W}}\gets \prod\limits_{j=1}^{n_W}\left(\textcolor{magenta}{g_1^\frac{\beta A_{p^{-1}(j+n_I)}(\tau)+\alpha B_{p^{-1}(j+n_I)}(\tau) + C_{p^{-1}(j+n_I)}(\tau)}{\delta}}\right)^{W_j}$
        \State $\rho, \sigma\probgets \mathbb{F}_q$
        \Statex
        \State Compute $\left(\textcolor{lightgray}{g_1^{A_j(\tau)}}\right)_{j=0}^{n_I+n_W}, \left(\textcolor{orange}{g_1^{B_j(\tau)}}\right)_{j=0}^{n_I+n_W}$ and $\left(\textcolor{RawSienna}{g_2^{B_j(\tau)}}\right)_{j=0}^{n_I+n_W}$ \Comment Using $CRS$. \footnotemark
        \Statex
        \State $g_1^A\gets \left(\textcolor{lime}{g_1^\delta}\right)^\rho\odot_{\mathbb{G}_1} \textcolor{red}{g_1^\alpha}\odot_{\mathbb{G}_1} \textcolor{lightgray}{g_1^{A_0(\tau)}}\odot_{\mathbb{G}_1} \prod\limits_{j=1}^{n_I+n_W}\left(\textcolor{lightgray}{g_1^{A_j(\tau)}}\right)^{S_j}$
        \State $g_1^B\gets \left(\textcolor{lime}{g_1^\delta}\right)^\sigma\odot_{\mathbb{G}_1} \textcolor{blue}{g_1^\beta} \odot_{\mathbb{G}_1} \textcolor{orange}{g_1^{B_0(\tau)}}\odot_{\mathbb{G}_1} \prod\limits_{j=1}^{n_I+n_W}\left(\textcolor{orange}{g_1^{B_j(\tau)}}\right)^{S_j}$
        \State $g_1^C\gets \textcolor{yellow}{g_1^W}\odot_{\mathbb{G}_1} \textcolor{NavyBlue}{g_1^S}\odot_{\mathbb{G}_1} \left(g_1^{A}\right)^\sigma\odot_{\mathbb{G}_1} \left(g_1^B\right)^\rho\odot_{\mathbb{G}_1} \left(\textcolor{lime}{g_1^\delta}\right)^{-\sigma\rho}$
        \State $g_2^B\gets \left(\textcolor{violet}{g_2^\delta}\right)^\sigma\odot_{\mathbb{G}_2} \textcolor{olive}{g_2^\beta} \odot_{\mathbb{G}_2} \textcolor{RawSienna}{g_2^{B_0(\tau)}}\odot_{\mathbb{G}_2}\prod\limits_{j=1}^{n_I+n_W}\left(\textcolor{RawSienna}{g_2^{B_j(\tau)}}\right)^{S_j}$
        \Statex
        \State $\pi\gets (g_1^A,g_1^C,g_2^B)$
        \State \Return $\pi$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}
\footnotetext{
    \label{footnote:qapPolynomialsAtTau}
    Let $g\in\{g_1, g_2\}$, let $P\in\bigcup\limits_{j=0}^{n_I+n_W}\{A_j, B_j\}$ with $P(x)=p_0x^0+p_1x^1+\dots+p_{n_P}x^{n_P}$. 
    With $\mathbb{G}\in\{\mathbb{G}_1,\mathbb{G}_2\}$ we denote the group $g$ is a part of.
    We compute $g^{P(\tau)}$ via\\
    \begin{align*}
        g^{P(\tau)}&=g^{p_0\tau^0+p_1\tau^1+\dots+p_{n_P}\tau^{n_P}}\\
        &=g^{p_0\tau^0}\odot_{\mathbb{G}}g^{p_1\tau^1}\odot_{\mathbb{G}}\dots \odot_{\mathbb{G}}g^{p_{n_P}\tau^{n_P}}\\
        &=\left(g^{\tau^0}\right)^{p_0}\odot_{\mathbb{G}} \left(g^{\tau^1}\right)^{p_1}\odot_{\mathbb{G}} \dots \odot_{\mathbb{G}}\left(g^{\tau^{n_P}}\right)^{p_{n_P}}
    \end{align*}
    Note that $\forall j\in\{0,\dots, n_P\}: g^{\tau^j}\in CRS$, since $n_P=deg(P)<k=deg(T)$.
    Therefore, computing $\left(g_1^{A_j(\tau)}\right)_{j=0}^{n_I+n_W}, \left(g_1^{B_j(\tau)}\right)_{j=0}^{n_I+n_W}$ and $\left(g_2^{B_j(\tau)}\right)_{j=0}^{n_I+n_W}$ from $CRS$ is possible.    
}
