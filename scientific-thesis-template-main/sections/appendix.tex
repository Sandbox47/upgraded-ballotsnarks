\begin{definition}[Bilinear Pairing \cite{least_authority_moonmath_2024}]
    \label{def:bilinearPairing}
    Let $(\mathbb{G}_1, \odot_{\mathbb{G}}), (\mathbb{G}_2, \odot_{\mathbb{G}_2})$ and $(\mathbb{G}_3, \odot_{\mathbb{G}_2})$ be groups.
    Then, a bilinear pairing on these groups is a function 
    \begin{align*}
        e:\mathbb{G}_1\times\mathbb{G}_2\to\mathbb{G}_3\text{,}
    \end{align*}
    where the bilinearity property holds:
    \begin{align*}
        &\forall g_1, g_1'\in\mathbb{G}_1\forall g_2, g_2'\in\mathbb{G}_2: \\
        &\quad e(g_1\odot_{\mathbb{G}_1}g_1', g_2)=e(g_1, g_2)\odot_{\mathbb{G}_3} e(g_1',g_2) \text{ and }\\
        &\quad e(g_1, g_2\odot_{\mathbb{G}_2}g_2')=e(g_1, g_2)\odot_{\mathbb{G}_3} e(g_1,g_2') \\
    \end{align*}
    $e$ is called non-degenerate, if:
    \begin{align*}
        &\forall g_1\in\mathbb{G}_1\forall g_2\in\mathbb{G}_2:
        &\quad e(g_1, g_2)=e_{\mathbb{G}_3}\quad\Leftrightarrow\quad g_1=e_{\mathbb{G}_1} \lor g_2=e_{\mathbb{G}_2}
    \end{align*}
    Here, $e_{\mathbb{G}_1}, e_{\mathbb{G}_2}$ and $e_{\mathbb{G}_3}$ denote the neutral elements of the respective groups.
\end{definition}

\section{Utility Circuits}
We use several utility circuits as modlues inside the ballot validity circuit $\mathfrak{C}_C^{Ballot\text{-}Validity}$.

\subsection{$reverse$}
\label{section:appendix:reverse}
To reduce visual clutter it is sometimes benefitial to reverse the order of inputs to a circuit.
Therefore, we use, \begin{align*}
    r_{reverse(l)}:\mathbb{F}^l\to\mathbb{F}^l, (x_0,\dots,x_{l-1})\to (x_{l-1}, \dots, x_0)
\end{align*}
and the corresponding circuit $\mathfrak{C}_{reverse(l)}$ \cref{fig:circuit:reverse}.
Note that $\mathfrak{C}_{reverse(l)}$ does not need any constraints and we use it purely for convenience.

\begin{figure}[h]
    \centering
    \includegraphics[scale=0.7]{graphics/appendix/reverse.pdf}
    \caption{Circuit $\mathfrak{C}_{reverse(l)}$ to reverse the order of $l$ signals.}
    \label{fig:circuit:reverse}
\end{figure}

\subsection{$Is\text{-}Equal$}
\label{section:appendix:isEqual}
In many cases, we need to compute whether two signals $x$ and $y$ are equal.
Therefore, we use, \begin{align*}
    r_{If\text{-}Equal}:\mathbb{F}^2\to\mathbb{F}, (x,y)\to \begin{cases}
        1,\quad \text{if } x=y\\
        0,\quad\text{otherwise}
    \end{cases}
\end{align*}
and the corresponding circuit $\mathfrak{C}_{Is\text{-}Equal}$ \cref{fig:circuit:isEqual}.
In our implementation, we use the template \verb|IsEqual| from the file \verb|comparators.circom| in the circomlib library \cite{jordy_baylina_iden3circomlib_2025}.

\begin{figure}[h]
    \centering
    \includegraphics[scale=0.7]{graphics/appendix/isEqual.pdf}
    \caption{Circuit $\mathfrak{C}_{Is\text{-}Equal}$ to compute whether two signals are equal.}
    \label{fig:circuit:isEqual}
\end{figure}

\subsection{$If\text{-}Else(l)$}
\label{section:appendix:ifElse}
Branching is very important for most use cases.
Often, the result of a computation depends on a condition.
For this purpse, we use, \begin{align*}
    r_{If\text{-}Else(l)}:&\mathbb{F}^l\times \mathbb{F}^l\times\mathbb{F}\to\mathbb{F}^l, (v_{else},v_{if}, cond)\to \begin{cases}
        v_{else},\quad \text{if } cond=0\\
        v_{if},\quad\text{if } cond=1\\
        undefined,\quad\text{otherwise}
    \end{cases} \\
    &\quad\text{with }\begin{matrix}
        v_{else} & = (v_0^{else}, \dots, v_{l-1}^{else})\\
        v_{if} & = (v_0^{if}, \dots, v_{l-1}^{if})
    \end{matrix}
\end{align*}
and the corresponding circuit $\mathfrak{C}_{If\text{-}Else}$ \cref{fig:circuit:ifElse}.

\begin{figure}[h]
    \centering
    \includegraphics[scale=0.7]{graphics/appendix/ifElse.pdf}
    \caption{Circuit $\mathfrak{C}_{If\text{-}Else(l)}$ which outputs either $v_{else}$ or $v_{if}$ depending on $cond$.
    Note that the circuit can produce arbitrary outputs if $cond\notin\{0,1\}$.}
    \label{fig:circuit:ifElse}
\end{figure}

\subsection{$Switch\text{-}Case(c)$}
\label{section:appendix:switchCase}
Branching is very important for most use cases.
Often, the result of a computation depends on a condition that can take several different values with a default value if none of the allowed values are chosen.
For this purpse, we use, \begin{align*}
    r_{Switch\text{-}Case(c)}:&\mathbb{F}^c\times \mathbb{F}^{c-1}, \\
    &(v, cond)\to \begin{cases}
        v_{i},\quad \text{if } cond_i=1 \land \forall j\in\{0,\dots,c-2\}\setminus\{i\}: cond_j=0\\
        v_{def},\quad\text{if } \forall i\in\{0,\dots, c-2\}:c_i=0\\
        undefined,\quad\text{otherwise}
    \end{cases} \\
    &\quad\text{with }\quad\begin{matrix}
        v &= (v_0, \dots, v_{c-2}, v_{def})\\
        cond & = (cond_0, \dots, cond_{c-2})
    \end{matrix}
\end{align*}
and the corresponding circuit $\mathfrak{C}_{Switch\text{-}Case}$ \cref{fig:circuit:switchCase}.

\begin{figure}[h]
    \centering
    \includegraphics[scale=0.7]{graphics/appendix/switchCase.pdf}
    \caption{Circuit $\mathfrak{C}_{Switch\text{-}Case(l)}$ which outputs the selected value from $v$ depending on $cond$.
    Note that the circuit can produce arbitrary outputs if \\$cond\notin\left\{(cond_0,\dots, cond_{c-2})\in\{0,1\}^{c-1}\Bigg|\sum_{i=0}^{c-2}cond_i\in\{0,1\}\right\}$.}
    \label{fig:circuit:switchCase}
\end{figure}

\subsection{$Num2Bits$}
\label{section:appendix:num2Bits}
In many cases, we need to compute the binary representation of a number $k$.
For this purpose, we use \begin{align*}
    r_{Num2Bits(l)}:\mathbb{F}\to\mathbb{F}^l, k\to (k_0,\dots, k_{l-1}) \text{ with } k=\sum_{i=0}^{l-1} k_i\cdot 2^i
\end{align*}
and the corresponding circuit $\mathfrak{C}_{Num2Bits(l)}$ \cref{fig:circuit:num2Bits}.
In our implementation, we use the template \verb|Num2Bits(l)| from the file \verb|bitify.circom| in the circomlib library \cite{jordy_baylina_iden3circomlib_2025}.

\begin{figure}[h]
    \centering
    \includegraphics[scale=0.7]{graphics/appendix/num2Bits.pdf}
    \caption{Circuit $\mathfrak{C}_{Num2Bits(l)}$ to compute the binary representation of a number $k$ with at most $l$ bits.}
    \label{fig:circuit:num2Bits}
\end{figure}
