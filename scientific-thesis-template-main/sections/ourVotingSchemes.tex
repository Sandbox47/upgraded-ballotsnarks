\chapter{Our Voting Schemes}
\label{chap:ourVotingSchemes}

The voting schemes we use, are built from the concepts introduced in the Chapters \ref{chap:gpzkp}, \ref{chap:elGamalEncryption} and \ref{chap:ellipticCurves}.
Recall, that we introduced a voting scheme as the following seve-tuple in \cref{chap:votingschemes}:
\begin{align*}
    VS=(C, \mathcal{S}, ZPS(R_C), Setup, \{V_0, \dots, V_{n-1}, VA\})
\end{align*}
We employ \gls{eeg} based on either a Montgomery Curve or its Twisted Edwards equivalent as the encryption scheme $\mathcal{S}$.
Furthermore, we use a \gls{g16snark} as the \gls{zps} to ensure ballot validity.
To show that our approach to voting schemes can be easily adapted to many different election types, we create voting schemes based on several different choice spaces $C$.

\section{The Ballot Validity Relation $R_C$}
\label{section:ourVotingSchemes:ballotValidity}

%    (And note that the decomposition relation based on this circuit is equivalent to the relation $R_C$ used in \cref{chap:votingschemes}.)

In \cref{chap:votingschemes}, $R_C$ is defined as the relation checking ballot validity.
Here, $R_C$ gets a ballot $b$ and auxiliary information $aux=(aux^{Vot}, aux^{Enc})$ as a private witness.
The public instance is the public key $pk$ of the encryption scheme and an encrypted ballot $b'$.
As presented earlier, $R_C$ outputs $1$ if and only if two checks are fulfilled:
\begin{itemize}
    \item $b\in C$
    \item $\mathcal{S}.Enc^{aux^{Enc}}(b)=b'$
\end{itemize}
In order to compute a \gls{zkp} for these checks, we need to express them inside a circuit.
The algebraic function asserting $b\in C$ is called $r_C^{Vot}$ and the corresponding circuit is $\mathfrak{C}_C^{Vot}$.
How we can compute $r_C^{Vot}$ is highly dependent on the choice space $C$ and therefore also the election type.
Thus, we will define this in detail for the different election types in \cref{section:ourVotingSchemes:choiceSpaces}.

For the assertion $\mathcal{S}.Enc^{aux^{Enc}}(b)=b'$ we create an algebraic function $r_C^{Enc}$ and corresponding circuit $\mathfrak{C}_C^{Enc}$.
This function depends on the choice of the encryption scheme $\mathcal{S}$ and will be defined in \cref{section:ourVotingSchemes:encryption}.

We can combine $r_C^{Vot}$ and $r_C^{Enc}$ into the function $r_C^{Ballot\text{-}Validity}$ which can be computed as in \cref{alg:ballotValidity}.

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Function $r_C^{Ballot\text{-}Validity}$ used to assert Ballot validity. 
    Input is a ballot $b$, encrypted ballot $b'$, public key $pk$ and auxiliary encryption information $aux$.
    %$r^{Ballot\text{-}Validity}$ asserts $b\in C$ via $r^{Vot}$ and $\mathcal{S}.Enc^{aux}(b)=b'$ via $r^{Enc}$.
    }
    \label{alg:ballotValidity}
    \begin{algorithmic}
    \Procedure{$r_C^{Ballot\text{-}Validity}$}{$pk, b, aux, b'$}
        \State $(aux^{Vot}, aux^{Enc})\gets aux$
        \State $r_C^{Vot}(b)$ \Comment Assert $b\in C$.
        \State $r_C^{Enc}(pk, b, aux^{Enc}, b')$ \Comment Assert $\mathcal{S}.Enc^{aux^{Enc}}(b)=b'$.
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

The corresponding circuit $\mathfrak{C}_C^{Ballot\text{-}Validity}$ is presented in \cref{fig:circuit:ballotValidity}.

    \caption{Circuit $\mathfrak{C}_C^{Ballot\text{-}Validity}$, asserting Ballot Validity.
    Input is \textcolor{witnessorange}{witness} and \textcolor{instanceblue}{instance}. 
    The witness consists of a ballot $b$ and auxiliary information $aux=(aux^{Vot}, aux^{Enc})$.
    $aux^{Enc}$ is typically the randomness needed for encrypting the individual ballot entries.

    The instance consists of an encrypted ballot $b'$ and the public key $pk$.

\end{figure}

Let $DR(QAP(R1CS(r_C^{Ballot\text{-}Validity})), d_p)$ be the decomposition relation which assigns $pk$ and $b'$ to the instance and $aux$ and $b$ to the witness as shown in \cref{fig:circuit:ballotValidity}.
Then, $R_C=DR(QAP(R1CS(r_C^{Ballot\text{-}Validity})), d_p)$.

\section{Zero Knowledge Proof System $ZPS(R_C)$}
\label{section:ourVotingSchemes:zps}
Assume, we have a ballot validity relation $R_C$ as introduced in \cref{chap:votingschemes}.
Moreover, assume that $R_C$ is a decomposition relation $DR(QAP(R1CS(r_C^{Ballot\text{-}Validity})), d_p)$ for the algebraic relation $r_C^{Ballot\text{-}Validity}$ as previously (\cref{section:ourVotingSchemes:ballotValidity}).
Then, we use the \gls{g16snark} $GSNARK(R_C)=(Setup, Prove, Verify)$ based on paramters $param=(\mathbb{G}_1, \mathbb{G}_2, e(\cdot, \cdot), g_1, g_2, \mathbb{F}_q)$ as $ZPS(R_C)$.
We use the Barreto-Naehrig curve \verb|BN254| \cite{todo cite} to instantiate $\mathbb{G}_1$ and $\mathbb{G}_2$.  
This curve has the order 
\begin{align*}
    \scriptstyle q = 21888242871839275222246405745257275088548364400416034343698204186575808495617
\end{align*}
which is used to define $\mathbb{F}_q$.
Therefore, $R_C$ operates over elements of $\mathbb{F}_q$.
For the bilinear pairing $e(\cdot, \cdot)$, we choose the Weil pairing or a variant of this. \cite{huber_zk-snarks_2025}

\section{Encryption Scheme $\mathcal{S}$}
\label{section:ourVotingSchemes:encryption}
We use an \gls{eeg} encryption scheme $\mathcal{S}$ because it is homomorphic and therefore allows us to aggregate ballots in their encrypted form (\cref{chap:elGamalEncryption}).
Furthermore, we choose an elliptic curve as the group $\mathbb{G}$ that the encryption scheme is based on for security reasons (\cref{chap:ellipticCurves}).

As established in \cref{section:ourVotingSchemes:zps}, our $GSNARK(R_C)$ requires that $R_C$ operates over $\mathbb{F}_q$ with 
\begin{align*}
    \scriptstyle q = 21888242871839275222246405745257275088548364400416034343698204186575808495617\text{.}
\end{align*}
Therefore, as the elliptic curve $\mathbb{G}$ used in $\mathcal{S}$, we need to choose one that operates over elements of $\mathbb{F}_q$.
Here, we implement two options: 
Either, we use the Montgomery Curve $M_{126932, 1}(\mathbb{F}_qP^2)$ or the isomorphic Twisted Edwards Curve $TE_{126934, 126930}(\mathbb{F}_q)$.\footnote{
    \label{footnote:ellipticCurveSubgroup}
    Note that the use of $M_{126932, 1}(\mathbb{F}_qP^2)$ or $TE_{126934, 126930}(\mathbb{F}_q)$ directly is insecure.
    These groups have the order $2^3\cdot p$ where
    \begin{align*}
        \scriptstyle p = 2736030358979909402780800718157159386074658810754251464600343418943805806723
    \end{align*}
    is the large prime factor.
    Then, computations in the subgroup of order $2^3=8$ are trivially insecure.
    Computing the discrete logarithm in this subgroup is efficiently possible by trying all possibilities.
    To avoid operating in this small subgroup, we actually choose the subgroup with order $p$ as the basis for \gls{eeg} encryption.
    Since all operations are defined the same as in the full group $M_{126932, 1}(\mathbb{F}_qP^2)$ or $TE_{126934, 126930}(\mathbb{F}_q)$, we abstract away from the details here.
}
This curve is an instantiation of Curve25519 \cite{curve25519}.
Huber et. al. \cite{huber_zk-snarks_2025} used the Montgomery Curve in their implementation.
However, as discussed in \cref{chap:ellipticCurves}, Twisted Edwards Curves can offer more efficient group exponentiation if we provide a list of powers of the element to exponentiate.

This leads to the two different choices $\mathcal{S}=EEG(M_{126932,1}(\mathbb{F}_qP^2))$ or $\mathcal{S}:=PPEEG(TE_{126934, 126930}(\mathbb{F}_qP^2))$. \cref{footnote:ellipticCurveSubgroup}
The corresponding instantiations of the circuit $\mathfrak{C}_C^{Enc}$ are illustrated in \cref{fig:circuit:encMEEG} and \cref{fig:circuit:encTEPPEEG}.\footnote{
    \label{footnote:encryptionComponentwiseCircuit}
    Note that encryption of a ballot $b\in C$ is done componentwise.
    Therefore, the encryption circuit $\mathfrak{C}_C^{Enc}$ only depends on the dimensions of $C$ but not the precise definition of $C$.
    %The definition of $\mathfrak{C}_C^{Enc}$ given here is for vector choice spaces ($\exists i: C\subseteq \mathbb{F}^i$).
    The definition of $\mathfrak{C}_C^{Enc}$ given here is for a choice space $C\subseteq \mathbb{F}$.
    For vector choice spaces ($\exists i: C\subseteq \mathbb{F}^i$), we have $i$ entries in $b$ and $aux^{Enc}$, and $i$ tuples in $b'$.
    Then, $\mathfrak{C}_C^{Enc}$ needs $i$ instantiations of $\mathcal{S}=EEG(M_{126932,1}(\mathbb{F}_qP^2))$ or $\mathcal{S}:=PPEEG(TE_{126934, 126930}(\mathbb{F}_qP^2))$ respectively to assert correct encryption of all entries.
    The definition for matrix choice spaces ($\exists i, j\in\mathbb{N}: C\subseteq \mathbb{F}^{i\times j}$) is analogous.
}

\section{Choice Spaces $C$}
\label{section:ourVotingSchemes:choiceSpaces}
The choice space $C$ in a voting system $VS(C)$ only influences the relation $R_C$ and by extension the \gls{zps} instantiation but not any other part of the voting scheme.
As described in \cref{section:ourVotingSchemes:ballotValidity}, there is an algebraic function $r_C$ and a function $d_p$ for which $R_C=DR(QAP(R1CS(r_C^{Ballot\text{-}Validity})), d_p)$.
Here, $r_C^{Ballot\text{-}Validity}$ can be split into a part $r_C^{Enc}$ for ensuring correct encryption and a part $r_C^{Vot}$ for ensuring that a ballot $b$ is part of $C$ (\cref{section:ourVotingSchemes:encryption}).
Since encryption is done component-wise, $r_C^{Enc}$ is only influenced by the dimensions of $C$ \cref{footnote:encryptionComponentwiseCircuit}.
Therefore, to instantiate our voting system $VS(C)$ for a new election type, we only need to define the choice space $C$ and the corresponding algebraic function $r_C^{Vot}$ for this election type.

As in \cite{huber_zk-snarks_2025}, we cover a variety of election types ranging from simple Single-Vote to more complicated ranking-based types like Condorcet.


\subsection{Single-Vote}
In a Single-Vote election, a voter can choose from a fixed set of candidates. 
Here, each voter can vote for at most one of the candidates.
Therefore, it is also possible to abstain from the vote.
The example voting system introduced in \cref{chap:votingschemes} is an election of this type.

\begin{definition}[Single-Vote System]
    Let $cand\in\mathbb{N}$ be the number of candidates, let $\mathbb{F}_q$ be a field.
    Then, the corresponding Single-Vote choice space is defined as follows:
    {\setlength{\mathindent}{2pt}
    \begin{align*}
        C_{\text{Single}(cand)}= \left\{(b_0,\dots, b_{cand-1})\in \{0,1\}^{cand}\Bigg| \sum_{i=0}^{cand-1}b_i\in\{0,1\}\right\}
    \end{align*}
    }
    
    To assert choice space membership, we employ the relation $r_{C_{\text{Single}(cand)}}^{Vot}:\mathbb{F}_q^{cand}\to\emptyset$ (\cref{alg:singleVoteVoting}) and corresponding circuit $\mathfrak{C}_{C_{\text{Single}(cand)}}^{Vot}$ (\cref{fig:circuit:singleVoteVoting}).\\
    Here, we assert that all individual entries of a ballot $b\in\mathbb{F}_q^{cand}$ are bits and that the sum of all entries is a bit. 
\end{definition}

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Function $r_{C_{\text{Single}(cand)}}^{Vot}$ used to assert that a ballot $b\in\mathbb{F}_q^{cand}$ is part of the choice space $C_{\text{Single(cand)}}$.}
    \label{alg:singleVoteVoting}
    \begin{algorithmic}
    \Procedure{$r_{C_{\text{Single}(cand)}}^{Vot}$}{$b$}
        \State $(b_0,\dots, b_{cand-1}):=b$
        \State $sum\gets 0$
        \For{$i\in\{0,\dots, cand-1\}$}
            \State $r_{assert\text{-}Bit}(b_i)$
            \State $sum \gets sum + b_i$
        \EndFor
        \State $r_{assert\text{-}Bit}(sum)$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

\subsection{Multi-Vote}
In a Multi-Vote election, a voter can choose from a fixed set of candidates. 
In contrast to Single-Vote, each voter can allocate up to a maximum of $max$ votes among the candidates, where each candidate can receive at most $t$ votes.

\begin{definition}[Multi-Vote System]
    Let $cand\in\mathbb{N}$ be the number of candidates, let $\mathbb{F}_q$ be a field.
    Let $max$ be the maximum number of votes each voter has and let $t$ be the maximum number of votes a voter can give a single candidate.
    Then, the corresponding Multi-Vote choice space is defined as follows:
    {\setlength{\mathindent}{2pt}
    \begin{align*}
        C_{\text{Multi}(cand, max, t)}=\left\{(b_0,\dots, b_{cand-1})\in \{0,\dots, t\}^{cand}\Bigg|\sum_{i=0}^{cand-1}b_i\in\{0,\dots, max\}\right\}
    \end{align*}
    }
    To assert choice space membership, we employ the relation $r_{C_{\text{Multi}(cand, max, t)}}^{Vot}:\mathbb{F}_q^{cand}\to\emptyset$ (\cref{alg:multiVoteVoting}) and corresponding circuit $\mathfrak{C}_{C_{\text{Multi}(cand, max, t)}}^{Vot}$ (\cref{fig:circuit:multiVoteVoting}).\\
    Here, we assert that all individual entries of a ballot $b\in\mathbb{F}_q^{cand}$ are at most $t$ and that the sum of all entries is at most $max$.
\end{definition}

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Function $r_{C_{\text{Multi}(cand, max, t)}}^{Vot}$ used to assert that a ballot $b\in\mathbb{F}_q^{cand}$ is part of the choice space $C_{\text{Multi}(cand, max, t)}$.}
    \label{alg:multiVoteVoting}
    \begin{algorithmic}
    \Procedure{$r_{C_{\text{Multi}(cand, max, t)}}^{Vot}$}{$b$}
        \State $(b_0,\dots, b_{cand-1}):=b$
        \State $sum\gets 0$
        \For{$i\in\{0,\dots, cand-1\}$}
            \State $r_{assert\text{-}Leq}(b_i, t)$
            \State $sum \gets sum + b_i$
        \EndFor
        \State $r_{assert\text{-}Leq}(sum, max)$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

We present two modifications of Multi-Vote that introduce novel choice spaces: Line-Vote and \gls{mwr}.
Both of these were introduced by Huber et. al. \cite{huber_zk-snarks_2025}.
Therefore, there are no specialized \gls{zkp} for these election types yet and developing them would be relatively complicated.
On the other hand, our voting system can be easily adapted to Line-Vote and \gls{mwr}.

\subsubsection{Line-Vote}
In a Line-Vote election, a voter can choose from a fixed set of options.
% Here, we can understand Line-Vote System $VS_{C_{Line\text{-}Vote(cand)}}$ as a Multi-Vote 
As in Multi-Vote with $t=1$, each voter can give at most one vote to each of the options.
The total amount of votes submitted by a voter is only limited by the number of options.
However, as an additional restriction in Line-Vote, we require that all $1$-votes form a continuous line.

Therefore, we can also interpret a Line-Vote system $VS_{C_{Line(cand)}}$ as the Multi-Vote system $VS_{C_{Multi(cand, cand, 1)}}$ with additionally enforcing that all $1$-votes form a continous line.

\begin{definition}[Line-Vote System]
    Let $cand\in\mathbb{N}$ be the number of candidates, let $\mathbb{F}_q$ be a field.
    Then, the corresponding Line-Vote choice space is defined as follows:
    {\setlength{\mathindent}{2pt}
    \begin{align*}
        C_{\text{Line}(cand)}=& \left\{(b_0,\dots, b_{cand-1})\in \{0,1\}^{cand}\Bigg|\right. \forall i, j \in \{0,\dots, cand-1\}:\\
        & \quad \left.\vphantom{\Bigg|}(i<j\land b_i=b_j=1)\Rightarrow (\forall k\in\{i,\dots, j\}: b_k=1)\right\}
    \end{align*}
    }
    To assert choice space membership, we employ the relation $r_{C_{\text{Line}(cand)}}^{Vot}:\mathbb{F}_q^{cand}\to\emptyset$ (\cref{alg:lineVoteVoting}) and corresponding circuit $\mathfrak{C}_{C_{\text{Line}(cand)}}^{Vot}$ (\cref{fig:circuit:lineVoteVoting}).\\
    Here, we assert asserts that all individual entries of a ballot $b\in\mathbb{F}_q^{cand}$ are bits and that the number of times that a $1$-vote follows after a $0$-vote is at most $1$.
\end{definition}

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Function $r_{C_{\text{Line}(cand)}}^{Vot}$ used to assert that a ballot $b\in\mathbb{F}_q^{cand}$ is part of the choice space $C_{\text{Line}(cand)}$.}
    \label{alg:lineVoteVoting}
    \begin{algorithmic}
    \Procedure{$r_{C_{\text{Line}(cand)}}^{Vot}$}{$b$}
        \State $(b_0,\dots, b_{cand-1}):=b$
        \State $ind\gets b_0$ \Comment $ind$ counts how often a $1$-vote follows after a $0$-vote
        \State $r_{assert\text{-}Bit}(b_0)$
        \For{$i\in\{1,\dots, cand-1\}$}
            \State $r_{assert\text{-}Bit}(b_i)$
            \State $ind \gets ind + (b_i-b_{i-1})\cdot b_i$ \Comment $(b_i-b_{i-1})\cdot b_i = \begin{cases}
                1 \text{, if } b_{i-1}=0\land b_i=1\\
                0 \text{, otherwise}
            \end{cases}$
        \EndFor
        \State $r_{assert\text{-}Bit}(ind)$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

\subsubsection{Multi-Vote with Rules}
A \gls{mwr} election is a Multi-Vote election with additional constraints relating some of the votes in a ballot.
As an example and for comparability with the implementation of Huber et. al. \cite{huber_zk-snarks_2025}, we require that the product of the second and third vote is equal to the first vote.
This illustrates nicely that we can extend the existing Multi-Vote system easily by adding additional restrictions.

\begin{definition}[Multi-Vote with Rules System]
    Let $cand\geq 3$ be the number of candidates, let $\mathbb{F}_q$ be a field.
    Let $max$ be the maximum number of votes each voter has and let $t$ be the maximum number of votes a voter can give a single candidate.
    Then, the corresponding \gls{mwr} choice space is defined as follows:
    {\setlength{\mathindent}{2pt}
    \begin{align*}
        C_{\text{MWR}(cand, max, t)}=\left\{(b_0,\dots, b_{cand-1})\in C_{\text{Multi}(cand)}\Bigg| b_0=b_1\cdot b_2\right\}
    \end{align*}
    }
    To assert choice space membership, we employ the relation $r_{C_{\text{MWR}(cand, max, t)}}^{Vot}:\mathbb{F}_q^{cand}\to\emptyset$ (\cref{alg:mwrVoting}) and corresponding circuit $\mathfrak{C}_{C_{\text{MWR}(cand, max, t)}}^{Vot}$ (\cref{fig:circuit:mwrVoting}).\\
    Here, we utilize $r_{C_{\text{Multi}(cand, max, t)}}$ (\cref{alg:multiVoteVoting}) and asserts that the first entry in a ballot equals the product of the second and third entry.
\end{definition}

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Function $r_{C_{\text{MWR}(cand, max, t)}}^{Vot}$ used to assert that a ballot $b\in\mathbb{F}_q^{cand}$ is part of the choice space $C_{\text{MWR}(cand, max, t)}$.}
    \label{alg:mwrVoting}
    \begin{algorithmic}
    \Procedure{$r_{C_{\text{MWR}(cand, max, t)}}^{Vot}$}{$b$}
        \State $(b_0,\dots, b_{cand-1}):=b$
        \State $r_{C_{\text{Multi}(cand, max, t)}}(b)$ \Comment Utilize existing Multi-Vote assertion (\cref{alg:multiVoteVoting})
        \State $b_0\mbeq b_1\cdot b_2$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

\subsection{Borda Elections}
In the Borda election methods, voters arrange the candidates in order of preference.
The ballot is then computed from this ranking by assigning points to the candidates depending on how they are ranked.
The better a candidate is ranked, the more points they receive in the ballot. 
What \glqq better\grqq\ means in this context is defined formally in \cref{def:ranking}

Note that secret ranking is not part of the ballot, but can sometimes greatly accelerate the choice space membership assertion.
Therefore, it is provided to the circuit $\mathfrak{C}_C^{Vot}$ as the auxiliary voting information $aux_{Vot}$.

\begin{definition}[Ranking]
    \label{def:ranking}
    Let $cand$ be the number of candidates in a voting system opertaing on the field $\mathbb{F}$
    Then, a ranking in the same voting system is a list $r=(r_0,\dots, r_{cand-1})\in\mathbb{F}^{cand}$ with the following meaning:\\
    $\forall i, j\in\{0,\dots cand-1\}$:
    \begin{itemize}
        \item $r_i > r_j\:\Rightarrow$ The candidate at position $i$ is \textit{ranked worse} than the candidate at position $j$.
        \item $r_i = r_j\:\Rightarrow$ The candidate at position $i$ is \textit{ranked the same} than the candidate at position $j$.
        \item $r_i > r_j\:\Rightarrow$ The candidate at position $i$ is \textit{ranked better} than the candidate at position $j$.
    \end{itemize}
\end{definition}

\subsubsection{Pointlist-Borda}
In a Pointlist-Borda election, as is common in Borda elections, a voter creates a ranking of the fixed set of candidates.
Using this ranking, the candidates are then assigned points according to a fixed, ordered pointlist.
Here, the highest ranked candidate receives the most points, the second-highest ranked candidate receives the secend-most points and so on until all points from the pointlist have been assigned.
All remaining candidates receive $0$ points.
This forms the submitted ballot of the voter.

Note that in this election type, we do not actually need the ranking to assert choice space membership efficiently.
Assume, we have $cand$ many candidates and a pointlist of length $l$.
In a valid ballot, for every entry in the pointlist, there is a candidate having this number of points assigned to them.
Additionally, there are $cand-l$ many candidates receiving $0$ points.

\begin{definition}[Pointlist-Borda System]
    Let $\mathcal{L}=(p_0,\dots, p_{l-1})\in\mathbb{N}^l$ be a pointlist.
    Let $cand\geq l$ be the number of candidates, let $\mathbb{F}_q$ be a field.
    Then, the corresponding Pointlist-Borda choice space is defined as follows:
    {\setlength{\mathindent}{2pt}
    \begin{align*}
        C_{\text{Pointlist-Borda}(cand, l, \mathcal{L})}=& \left\{(b_0,\dots, b_{cand-1})\in \mathbb{F}_q^{cand}\Bigg| (\forall p\in\mathcal{L}\exists i\in\{0,\dots, cand-1\}: b_i=p)\right.\\
        &\quad\land \left. \vphantom{\Bigg|} |\{i\in \{0,\dots, cand-1\}|b_i=0\}|=cand-l\right\}
    \end{align*}
    }
    To assert choice space membership, we employ the relation $r_{C_{\text{Pointlist-Borda}(cand, l, \mathcal{L})}}^{Vot}:\mathbb{F}_q^{cand}\times\mathbb{F}_q^{cand}\to\emptyset$ (\cref{alg:pointlistBordaVoting}) and corresponding circuit $\mathfrak{C}_{C_{\text{Pointlist-Borda}(cand, l, \mathcal{L})}}^{Vot}$ (\cref{fig:circuit:pointlistBordaVoting}).\\
    Here, we assert that all points in the pointlist occur exactly once in a ballot $b\in\mathbb{F}_q^{cand}$ and that there are $cand-l$ many votes that are $0$.
    The ranking $r\in\mathbb{F}_q^{cand}$ that $b$ is computed from is not needed to assert choice space memebership efficiently.
\end{definition}

        \State $cand-l \mbeq r_{occurences(l)}(b, 0)$

\subsubsection{Borda Tournament Style}
In a \gls{bts} election, as is common in Borda elections, a voter creates a ranking of the fixed set of candidates.
Using this ranking, the candidates are then assigned a fixed number of points for every candidate ranked worse than them and a different, fixed, and lower number of points for every candidate ranked the same as them.

\begin{definition}[Borda Tournament Style System]
    Let $cand\in\mathbb{N}$ be the number of candidates, let $\mathbb{F}_q$ be a field.
    Let $ap$ be the number of points to be given to a candidate for every other candidate ranked worse than them.
    Similarly, let $bp$ be the number of points to be given to a candidate for every other candidate ranked the same as them.
    Then, the corresponding \gls{bts} choice space is defined as follows:
    %\todo{Choice space definition}
    {\setlength{\mathindent}{2pt}
    \begin{align*}
        C_{\text{BTS}(cand, ap, bp)}=& \left\{(b_0,\dots, b_{cand-1})\in \mathbb{F}_q^{cand}\Bigg| \right.\\
        & \quad\exists (r_0,\dots, r_{cand-1})\in\mathbb{F}_q^{cand}\:\forall i\in\{0,\dots, cand-1\}:\\
        & \quad\quad b_i=ap\cdot |\{j\in\{0,\dots, cand-1\}|r_j>r_i\}| \\
        & \quad\quad \phantom{b_i=}\left. \vphantom{\Bigg|}+ bp\cdot |\{j\in\{0,\dots, cand-1\}\setminus\{i\}|r_j=r_i\}|\right\}
    \end{align*}
    }
    To assert choice space membership, we employ the relation $r_{C_{\text{BTS}(cand, ap, bp)}}^{Vot}:\mathbb{F}_q^{cand}\times\mathbb{F}_q^{cand}\to\emptyset$ (\cref{alg:btsVoting}) and corresponding circuit $\mathfrak{C}_{C_{\text{BTS}(cand, ap, bp)}}^{Vot}$ (\cref{fig:circuit:btsVoting}).\\
    Here, we compute a ballot $b'$ from the ranking $r$ according to the parameters $ap$ and $bp$ using the function $r_{compute\text{-BTS}(cand, ap, bp)}:\mathbb{F}_q^{cand}\to\mathbb{F}_q^{cand}, r\mapsto b'$ (\cref{alg:btsComputeBallot}) and corresponding circuit $\mathfrak{C}_{compute\text{-BTS}(cand, ap, bp)}$ (\cref{fig:circuit:btsComputeBallot}).
    Then, we assert that the computed ballot $b'$ equals the submitted ballot $b$.
\end{definition}

    Here, we use $r_{compute\text{-BTS}(cand, ap, bp)}$ (\cref{alg:btsComputeBallot}) to compute the ballot $b'$ corresponding to the ranking $r$.
    }
    \label{alg:btsVoting}
    \begin{algorithmic}
    \Procedure{$r_{C_{\text{BTS}(cand, ap, bp)}}^{Vot}$}{$b, r$}
        \State $b'\gets r_{compute\text{-BTS}(cand, ap, bp)}(r)$
        \State $b'\mbeq b$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

    \caption{Function $r_{compute\text{-BTS}(cand, ap, bp)}$ used to compute the ballot $b'$ belonging to ranking $r$ in a \gls{bts} voting system with choice space $C_{\text{BTS}(cand, ap, bp)}$.

\subsection{Condorcet Methods}
Condorcet elections are ranking based.
As in Borda elections, a user creates a ranking (\cref{def:ranking}) of $cand$ many candidates.
The ballot is then a matrix $b\in\mathbb{F}_q^{cand\times cand}$ computed from this ranking.
Here, the entry at position $(i,j)$ of the matrix is $1$ if candiate $i$ is ranked better than candidate $j$ and $0$ otherwise.

For comparibility with the implementation of Huber et. al. \cite{huber_hicolasnuberballotsnarks_2024}, we implement a choice space membership assertion that does not use the ranking.
However, by utilizing the ranking, we can greatly improve the efficiency of this assertion.
Therefore, we implement this variant as well.

\begin{definition}[Condorcet System]
    Let $cand\in\mathbb{N}$ be the number of candidates, let $\mathbb{F}_q$ be a field.
    Then, the corresponding \gls{bts} choice space is defined as follows:
    %\todo{Choice space definition}
    {\setlength{\mathindent}{2pt}
    \begin{align*}
        C_{\text{Condorcet}(cand)}=& \left\{b=(b_{i,j})_{i,j\in\{0,\dots, cand-1\}}\in \{0,1\}^{cand\times cand}\Bigg| \right.\\
        & \quad\exists (r_0,\dots, r_{cand-1})\in\mathbb{F}_q^{cand}\:\forall i, j\in\{0,\dots, cand-1\}:\\
        & \quad\quad \left. \vphantom{\Bigg|} (r_i>r_j\Rightarrow b_{i,j}=0 \land b_{j,i} = 1)\:\land\:(r_i=r_j\Rightarrow b_{i,j}=b_{j,i}=0)\right\} \\
        %& \quad\quad \left. \vphantom{\Bigg|}\land r_i=r_j\Rightarrow b_{i,j}=b_{j,i}=0\right\}
    \end{align*}
    }
    To assert choice space membership, we employ the relation $r_{C_{\text{Condorcet}(cand)}}^{Vot}:\mathbb{F}_q^{cand\times cand}\times\mathbb{F}_q^{cand}\to\emptyset$ and corresponding circuit $\mathfrak{C}_{C_{\text{Condorcet}(cand)}}^{Vot}$.\\
    Here, we implement two options:
    \begin{itemize}
        \item Without using the ranking (\cref{alg:condorcetVotingNoRanking}, \cref{fig:circuit:condorcetVotingNoRanking}):
        We assert that all entries of the submitted ballot $b$ are bits and that for $i\neq j$ also $b_{i,j}+b_{j,i}$ is a bit.
        This covers the cases \glqq$r_i>r_j$\grqq\ and \glqq$r_j=r_i$\grqq\ from the choice space definition. 
        Additionally, we need to show that $b$ belongs to a ranking $r$ in which transitivity holds.
        We can achieve this by computing a check matrix $a=(a_{i,j})_{i,j\in\{0,\dots, cand-1\}}:=(1-b_{i,j})_{i,j\in\{0,\dots cand-1\}}$ and asserting that $a_{i,j}\cdot a_{j,k}\cdot 1-b_{i,k}=0$ for all triples of distinc values $(i,j,k)\in\{0,\dots, cand-1\}^3$ \cite{huber_zk-snarks_2025}.
        \item Utilizing the ranking (\cref{alg:condorcetVotingWithRanking}, \cref{fig:circuit:condorcetVotingWithRanking}): 
        We compute a ballot $b'$ from the ranking $r$ using the function $r_{compute\text{-Condorcet}(cand)}:\mathbb{F}_q^{cand}\to\mathbb{F}_q^{cand\times cand}, r\mapsto b'$ (\cref{alg:condorcetComputeBallot}) and corresponding circuit $\mathfrak{C}_{compute\text{-Condorcet}(cand)}$ (\cref{fig:circuit:condorcetComputeBallot}).
        Then, we assert that the computed ballot $b'$ equals the submitted ballot $b$.
    \end{itemize}
\end{definition}

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Function $r_{C_{\text{Condorcet}(cand)}}^{Vot}$ used to assert that a ballot $b\in\mathbb{F}_q^{cand\times cand}$ computed from a ranking $r\in\mathbb{F}_q^{cand}$ is part of the choice space $C_{\text{Condorcet}(cand)}$.
    Here, we show the version that does not use the ranking $r$.
    }
    \label{alg:condorcetVotingNoRanking}
    \begin{algorithmic}
    \Procedure{$r_{C_{\text{Condorcet}(cand)}}^{Vot}$}{$b, r$}
        \State $(b_{i,j})_{i,j\in\{0,\dots, cand-1\}}:=b$
        \Statex
        \For{$i\in\{0,\dots, cand-1\}$}
            \For{$j\in\{0,\dots, cand-1\}$}
                \State $r_{assert\text{-}Bit}(b_{i,j})$
            \EndFor
        \EndFor
        \Statex
        \For{$i\in\{0,\dots, cand-1\}$}
            \For{$j\in\{i+1,\dots, cand-1\}$}
                \State $r_{assert\text{-}Bit}(b_{i,j} + b_{j,i})$
            \EndFor
        \EndFor
        \Statex
        \State $a=(a_{i,j})_{i,j\in\{0,\dots, cand-1\}}:=(1-b_{i,j})_{i,j\in\{0,\dots cand-1\}}$ \Comment Check matrix
        \For{$i\in\{0,\dots, cand-1\}$}
            \For{$j\in\{0,\dots, cand-1\}\setminus\{i\}$}
                \For{$k\in\{0,\dots, cand-1\}\setminus\{i, j\}$}
                    \State $a_{i, j}\cdot a_{j,k}\cdot (1-a_{i,k})\mbeq 0$
                \EndFor
            \EndFor
        \EndFor
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

\subsection{Majority Judgment}
In a Majority Judgment election, voters grade candidates independently by giving each candidate a grade from a predefined list of grades.
Assume, there are $cand$ many candidates and $grades$ many grades. 
Then, the submitted ballot of a voter is a matrix $b\in\mathbb{F}_q^{cand\times grades}$ where every row has exactly one $1$-entry in the column that corresponds to the grade given to that candidate.
All other entries are $0$.

\begin{definition}[Majority Judgment System]
    Let $cand\in\mathbb{N}$ be the number of candidates, let $grades\in\mathbb{N}$ be the number of grades and let $\mathbb{F}_q$ be a field.
    Then, the corresponding Majority Judgment choice space is defined as follows:
    {\setlength{\mathindent}{2pt}
    \begin{align*}
        C_{\text{Majority-Judgment}(cand, grades)}=& \left\{b=(b_{i,j})_{i\in\{0,\dots, cand-1\}, j\in\{0,\dots, grades-1\}}\in \{0,1\}^{cand\times grades}\Bigg| \right.\\
        &\quad\left.\vphantom{\Bigg|}\forall i\in\{0,\dots, cand-1\}: \sum_{j\in\{0,\dots, grades-1\}} b_{i,j}=1\right\}\\
    \end{align*}
    }
    To assert choice space membership, we employ the relation $r_{C_{\text{Majority-Judgment}(cand, grades)}}^{Vot}:\mathbb{F}_q^{cand}\times\mathbb{F}_q^{grades}\to\emptyset$ (\cref{alg:majorityJudgmentVoting}) and corresponding circuit $\mathfrak{C}_{C_{\text{Majority-Judgment}(cand, grades)}}^{Vot}$ (\cref{fig:circuit:majorityJudgmentVoting}).\\
    Here, we assert that all entries in the ballot $b$ are bits and that in every row of $b$, the sum of all entries is $1$.
\end{definition}

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Function $r_{C_{\text{Majority-Judgment}(cand, grades)}}^{Vot}$ used to assert that a ballot $b\in\mathbb{F}_q^{cand\times grades}$ is part of the choice space $C_{\text{Majority-Judgment}(cand, grades)}$.
    Here, we use $r_{assert\text{-}Bit}$ as an auxiliary function.
    }
    \label{alg:majorityJudgmentVoting}
    \begin{algorithmic}
    \Procedure{$r_{C_{\text{Majority-Judgment}(cand, grades)}}^{Vot}$}{$b$}
        \State $(b_{i,j})_{i\in\{0,\dots, cand-1\}, j\in\{0,\dots, grades-1\}}:=b$
        \For{$i\in\{0,\dots, cand-1\}$}
            \For{$j\in\{0,\dots, grades-1\}$}
                \State $r_{assert\text{-}Bit}(b_{i,j})$
            \EndFor
            \State $1\mbeq \sum_{j=0}^{grades-1} b_{i,j}$
        \EndFor
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}
