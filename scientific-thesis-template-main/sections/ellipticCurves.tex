\chapter{Elliptic Curves}
\label{chap:ellipticCurves}

In our voting scheme, we use \gls{eeg} encryption for the ballots.
All operations in the encryption of a value in \gls{eeg} are performed in a finite cyclic group $\mathbb{G}$.
For a secure encryption, the \gls{ddh} assumption needs to hold on $\mathbb{G}$.
Elliptic curves of a sufficiently large order $q$ are finite cyclic groups on which the \gls{ddh} assumption is believed to hold.
Moreover, Elliptic curves are benefitial compared to other cyclic groups because they need a smaller order to produce similar levels of security \cite{noauthor_case_2009}.

Fundamentally, Elliptic curves are sets of points that satisfy certain equations \cite{least_authority_moonmath_2024}. 
Here, a curve point consists of multiple elements from a so-called base field $\mathbb{F}_p$ of order $p$.
For better distinction, we call the field $\mathbb{F}_q$ with elliptic curve order $q$ the scalar field of the elliptic curve.

Our introduction of elliptic curves largely parallels the one given in chapter 5 of \cite{least_authority_moonmath_2024}.
However, as we exclusively use Montgomery curves and Twisted Edwards curves, we introduce them directly and do not introduce Short Weierstrass Curves seperately.

\section{Montgomery Curves}
\label{section:ellipticCurves:montgomery}
The most general form of Elliptic curves are so-called Short Weierstrass curves \cite{least_authority_moonmath_2024}.
Montgomery Curves are a subclass of Short Weierstrass curves \cite{montgomery_speeding_nodate}.
We can transform any Montgomery curve into an isomorphic Short Weierstrass curve but not every Short Weierstrass curves has an isomorphic Montgomery counterpart.
Overall, Montgomery curves are more limited than Short Weierstrass curves \cite{least_authority_moonmath_2024}.

However, as established previously, we will later use an Elliptic curve as the group $(\mathbb{G}, \odot_{\mathbb{G}})$ for \gls{eeg} encryption.
In \gls{eeg} we need to compute exponentiations of group elements, i.e., $g^m:=\underbrace{g\odot_{\mathbb{G}}\dots\odot_{\mathbb{G}}g}_{m \text{ times}}$ very often.
Even if we do not provide precomputed powers of $g$, Montgomery curves still allow us to compute $g^m$ efficiently using the Montgomery Ladder algorithm.

\subsection{Affine Montgomery Form}
As in \cite{least_authority_moonmath_2024} we introduce elliptic curves first in their affine representation and then in their projective representation.
Later on, we use both representations in the voting schemes.

\begin{definition}[Affine Montgomery Curve]
    Let $(\mathbb{F}, +, \cdot)$ be a prime field of order $p>3$.
    Let $A, B\in\mathbb{F}$ be two field elements with $B\neq 0$ and $A^2\mod p\neq4$.
    Then, the corresponding Affine Montgomery curve is the group $(M_{A,B}(\mathbb{F}), \odot_M^{Aff})$, with:
    \begin{itemize}
        \item The set of points $M_{A,B}(\mathbb{F})$:
        \begin{align*}
            M_{A,B}(\mathbb{F})=\{(x,y)\in\mathbb{F}\times\mathbb{F}|B\cdot y^2=x^3+A\cdot x^2+x\}\cup \{\mathcal{O}\}
        \end{align*}
        Here, $\mathcal{O}$ is the so-called \glqq point at infinity\grqq.
        \item The group law $\odot_M^{Aff}$:\\
        Let $P_1, P_2\in M_{A,B}(\mathbb{F})$.
        Then:
        {\setlength{\mathindent}{0pt}
        \begin{align*}
            P_1\odot_M^{Aff} P_2 =\begin{cases}
                P_1, &\quad\text{if } P_2=\mathcal{O}\\
                P_2, &\quad\text{if } P_1=\mathcal{O}\\
                \mathcal{O}, &\quad\text{if } P_1=(x, y) \text{ and } P_2=(x, -y)\\
                TangentRule(x,y), &\quad\text{if } P_1=P_2=(x,y) \text{ and } y\neq 0\\
                ChordRule(x_1, y_1, x_2, y_2), &\quad\text{otherwise, with } P_1=(x_1, y_1), P_2=(x_2, y_2)
                %\left(\left(\frac{3x^2+a}{2y}\right)^2-2x, \left(\frac{3x^2+a}{2y}\right)\cdot\left(x-\left(\left(\frac{3x^2+a}{2y}\right)^2-2x\right)\right)-y\right), &\quad\text{if } P_1=P_2=(x,y) \text{ and } y\neq 0\\
                %\left(\left(\frac{y_2-y_1}{x_2-x_2}\right)^2-x_1-x_2, \left(\frac{y_2-y_1}{x_2-x_1}\right)\cdot\left(x_1-\left(\left(\frac{y_2-y_1}{x_2-x_2}\right)^2-x_1-x_2\right)\right)-y_1\right), &\quad\text{otherwise, with } P_1=(x_1, y_1), P_2=(x_2, y_2)
            \end{cases}
        \end{align*}
        }
        $TangentRule$ and $ChordRule$ are defined in \cref{alg:tangentRuleAffineMontgomery} and \cref{alg:chordRuleAffineMontgomery}.
    \end{itemize}
    The neutral element of $M_{A,B}(\mathbb{F})$ is $\mathcal{O}$.
    For $P\in M_{A,B}(\mathbb{F})$, the inverse element $P^{-1}$ is defined as $P^{-1}:=\begin{cases}
        \mathcal{O}, &\quad\text{if } P=\mathcal{O}\\
        (x, -y), &\quad\text{otherwise, with } P=(x,y)
    \end{cases}$.
\end{definition}

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Affine Tangent Rule used to compute $P\odot_M^{Aff} P$ where $P=(x,y)\in M_{A,B}(\mathbb{F})\setminus\{\mathcal{O}\}$ and $y\neq 0$.}
    \label{alg:tangentRuleAffineMontgomery}
    \begin{algorithmic}
    \Procedure{$TangentRule$}{$x,y$}
        \State $tmp \gets  \frac{3x^2+2Ax+1}{2By}$
        \State $x' \gets  tmp^2\cdot B-2x-A$
        \State $y'\gets  tmp\cdot (x-x')-y$
        \State \Return $(x', y')$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Affine Chord Rule used to compute $P_1\odot_M^{Aff} P_2$ where $P_1=(x_1,y_1), P_2=(x_2, y_2)\in M_{A,B}(\mathbb{F})\setminus\{\mathcal{O}\}$ and $x_1\neq x_2$.}
    \label{alg:chordRuleAffineMontgomery}
    \begin{algorithmic}
    \Procedure{$ChordRule$}{$x_1,y_1, x_2, y_2$}
        \State $tmp \gets  \frac{y_2-y_1}{x_2-x_1}$
        \State $x' \gets  tmp^2\cdot B-x_1-x_2-A$
        \State $y'\gets  tmp\cdot (x_1-x')-y_1$
        \State \Return $(x', y')$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

\subsection{Projective Montgomery Form}
To avoid the special point at infinity $\mathcal{O}$ there exists the projective representation of elliptic curves based on projective planes.

\begin{definition}[Projective Point and Projective Plane]
    Let $(\mathbb{F}, +, \cdot)$ be a field. 
    We define $\mathbb{F}^*$ as the set of invertible elements in $\mathbb{F}$.
    Let $(X, Y, Z)=\in\mathbb{F}^3$. 
    Then, the projective point corresponding to $(X, Y, Z)$ is defined as
    \begin{align*}
        [X:Y:Z] :=\{(k\cdot X, k\cdot Y, k\cdot Z)|k\in\mathbb{F}^*\}\text{.}
    \end{align*}
    The projective plane of $\mathbb{F}$ is defined as
    \begin{align*}
        \mathbb{F}P^2:=\{[X:Y:Z]|(X,Y,Z)\in\mathbb{F}^3 \text{ and } (X, Y, Z)\neq (0,0,0)\}\text{.}
    \end{align*}
\end{definition}

Using projective planes, we can define Montgomery Curves in their projective representation as groups of projective points.

\begin{definition}[Projective Montgomery Curve]
    Let $(\mathbb{F}, +, \cdot)$ be a prime field of order $p>3$. 
    Let $A, B\in\mathbb{F}$ be two field elements with $B\neq 0$ and $A^2\neq4 \mod p$.
    Let $\mathbb{F}P^2$ be the projective plane of $\mathbb{F}$.
    Then, the corresponding Projective Montgomery curve is the group $(M_{A,B}(\mathbb{F}P^2), \odot_M^{Pro})$, with:
    \begin{itemize}
        \item The set of points $M_{A,B}(\mathbb{F}P^2)$:
        \begin{align*}
            M_{A,B}(\mathbb{F}P^2)=\{[X:Y:Z]\in\mathbb{F}P^2|B\cdot Y^2\cdot Z=X^3+A\cdot X^2\cdot Z+X\cdot Z^2\}
        \end{align*}
        Here, $[0:1:0]$ corresponds to the point at infinity $\mathcal{O}$ of the affine form.
        \item The group law $\odot_M^{Pro}$:\\
        Let $P_1=[X_1:Y_1:Z_1], P_2=[X_2:Y_2:Z_2]\in M_{A,B}(\mathbb{F}P^2)$. 
        To compute $P_1\odot_M^{Pro} P_2$ we transform $P_1, P_2$ to corresponding points on the corresponding Affine Montgomery Curve $M_{A,B}(\mathbb{F})$, use the affine group law $\odot_{M}^{Aff}$ and transform the result back to $M_{A,B}(\mathbb{F}P^2)$.
        Formally:
        {\setlength{\mathindent}{0pt}
        \begin{align*}
            %P_1\odot_M^{Pro} P_2 = I(I^{-1}(P_1)\odot_M^{Aff}I^{-1}(P_2))
            P_1\odot_M^{Pro} P_2 =\begin{cases}
                P_1, &\quad\text{if } P_2=[0:1:0]\\
                P_2, &\quad\text{if } P_1=[0:1:0]\\
                [0:1:0], &\quad\text{if } [X_1:Y_1:Z_1]\\
                &\quad\phantom{\text{if }}\:=[X_2:-Y_2:Z_2]\\
                TangentRule([X_1:Y_1:Z_1]), &\quad\text{if } P_1=P_2 \text{ and } Y_1\neq 0\\
                ChordRule([X_1:Y_1:Z_1], [X_2:Y_2:Z_2]), &\quad\text{otherwise}
            \end{cases}
        \end{align*}
        }
        $TangentRule$ and $ChordRule$ are defined in \cref{alg:tangentRuleProjectiveMontgomery} and \cref{alg:chordRuleProjectiveMontgomery}.
        Here, we follow the implementation from \cite{farashahi_faster_2024}.
        % Here, $I$ converts points from affine representation to projective representation, and $I^{-1}$ executes the reverse conversion.
        % $I$ and $I^{-1}$ are introduced in \cref{sec:coordinateTranformation}.
    \end{itemize}
    The neutral element of $M_{A,B}(\mathbb{F}P^2)$ is $[0:1:0]$.
    For $P=[X:Y:Z]\in M_{A,B}(\mathbb{F}P)$, the inverse element $P^{-1}$ is defined as $P^{-1}:=[X:-Y:Z]$.
\end{definition}

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Projective Tangent Rule used to compute $P\odot_M^{Pro} P$ where $P=[X:Y:Z]\in M_{A,B}(\mathbb{F}P^2)\setminus\{[0:1:0]\}$.}
    \label{alg:tangentRuleProjectiveMontgomery}
    \begin{algorithmic}
    \Procedure{$TangentRule$}{$[X:Y:Z]$}
        \State $I\gets 3X^2+2AXZ+Z^2$
        \State $H\gets 2BYZ$
        \State $X'\gets BI^2HZ^2-(AZ^2+2XZ)H^3$
        \State $Y'\gets -BI^3Z^2+(AZ^2+3XZ)IH^2-YZH^3$
        \State $Z'\gets H^3Z^2$
        \State \Return $[X':Y':Z']$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Projective Chord Rule used to compute $P_1\odot_M^{Pro} P_2$ where $P_1=[X_1:Y_1:Z_1], P_2=[X_2:Y_2:Z_2]\in M_{A,B}(\mathbb{F}P^2)\setminus\{[0:1:0]\}$ and $P_1\neq P_2^{-1}$.}
    \label{alg:chordRuleProjectiveMontgomery}
    \begin{algorithmic}
    \Procedure{$ChordRule$}{$[X_1:Y_1:Z_1], [X_2:Y_2:Z_2]$}
        \State $I\gets Y_2Z_1-Y_1Z_2$
        \State $H\gets X_2Z_1-X_1Z_2$
        \State $X'\gets BI^2HZ_1Z_2-(AZ_1Z_2+X_1Z_2+X_2Z_1)H^3$
        \State $Y'\gets -BI^3Z_1Z_2+(AZ_1Z_2+2X_1Z_2+X_2Z_1)IH^2-Y_1Z_2H^3$
        \State $Z'\gets H^3Z^2$
        \State \Return $[X':Y':Z']$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

\subsection{Transforming between Projective and Affine Coordinates}
\label{sec:coordinateTranformation}
As mentioned earlier, we will use both the projective and affine representation of elliptic curves in our voting schemes.
Therefore, we need to be able to convert from an affine elliptic curve $M_{A,B}(\mathbb{F})$ to a corresponding projective elliptic curve $M_{A,B}(\mathbb{F}P^2)$ and vice versa.

\begin{definition}[Coordinate Transformation]
    \label{def:coordinateTransformation}
    Let $(\mathbb{F}, +, \cdot)$ be a finite field of order $q$ with $q>3$. 
    % Let $a,b\in \mathbb{F}$ with $4a^2 + 27b^2\neq 0$. 
    Let $\mathbb{F}P^2$ be the projective plane of $\mathbb{F}$.
    Then, the function $I:\mathbb{F}^2\cup\{\mathcal{O}\}\to\mathbb{F}P^2$ converts points from affine representation to projective representation.
    Consequently, $I^{-1}:\mathbb{F}P^2\to\mathbb{F}^2\cup\{\mathcal{O}\}$ converts points from projective representation to affine representation.
    Here, we define:
    \begin{align*}
        I(P) &:= \begin{cases}
            [x:y:1], \quad\text{if } P\neq\mathcal{O}, \text{ with } P=(x,y)\\
            [0:1:0], \quad\text{otherwise}
        \end{cases}\\
        I^{-1}([X:Y:Z]) &:= \begin{cases}
            \left(\frac{X}{Z}, \frac{Y}{Z}\right), \quad\text{if } Z\neq 0\\
            \mathcal{O}, \quad\text{otherwise}
        \end{cases}
    \end{align*}
\end{definition}

\subsection{Exponentiation}
We use a Montgomery curve $M_{A,B}(\mathbb{F}P^2)$ as group in \gls{eeg} encryption because computing exponentiations of group elements is comparatively efficient.
Group exponentiation is defined as follows:
\begin{align*}
    \forall P=[X:Y:Z]\in M_{A,B}(\mathbb{F}P^2)\forall k\in\mathbb{N}:\quad P^k:=\underbrace{P\odot_{M}^{Pro}\dots\odot_{M}^{Pro}P}_{k \text{ times}}
\end{align*}
However, computing $P^k=[X_k:Y_k:Z_k]$ by iteratively using the group law $\odot_{M}^{Pro}$ $k-1$ times is computationally expensive as discussed in \cref{section:elGamalEncryption:precomputedPowers}.
Therfore, we utilize the Montgomery Ladder algorithm \cite{costello_montgomery_2018} based on $FastExp$ presented in \cref{alg:fastExp}.
This algorithm requires $\mathcal{O}(\log_2(k))$ many iterations to compute the coordinates $X_k$ and $Z_k$ of $P^k$ from the coordinates $X$ and $Z$ of $P$.
Afterwards, we use the Okeya-Sakurai $Y$-Recovery algorithm \cite{goos_efficient_2001} to reconstruct $Y_k$ from the last two iterations of the Montgomery Ladder.

\subsubsection{Pseudooperations}
For the Montgomery Ladder algorithm we work only with the $X$ and $Z$ coordinates of points $P=[X:Y:Z]\in M_{A,B}(\mathbb{F}P^2)$.
For this purpose, we define a quotient map $x$ to map projective points to pairs of their $X$ and $Z$ coordinates.
Specially treated is the point at infinity $[0:1:0]$.
\begin{definition}[Quotient Map of Montgomery Curves]
    Let $M_{A,B}(\mathbb{F}P^2)$ be a Montgomery curve.
    Then, we define the quotient map $x$ as follows:
    \begin{align*}
        &x:M_{A,B}(\mathbb{F}P^2)\to \{[X:Z]|X,Z\in\mathbb{F}\},\\
        &x([X:Y:Z]) :=\begin{cases}
            [X:Z], \quad\text{if } Z\neq 0\\
            [1:0], \quad\text{otherwise ($\rightsquigarrow [X:Y:Z]=[0:1:0]$)}
        \end{cases}
    \end{align*}
    We call $x(P)$ the Pseudopoint of $P\in M_{A,B}(\mathbb{F}P^2)$.
\end{definition}

Using this map, we want to be able to apply the group law to Pseudopoints. 
In other words, we want to compute $x(P\odot_M^{Pro}Q)$ from $x(P)$ and $x(Q)$.
Unfortunately, as mentioned in chapter 3.1 of \cite{costello_montgomery_2018} this is not possible, since $x(P)$ and $x(Q)$ do not uniquely determine $x(P\odot_M^{Pro} Q)$.
However, it is possible to determine $x(P\odot_M^{Pro} Q)$ based on $x(P), x(Q)$ and $x(P\odot_M^{Pro}Q^{-1})$.
This is useful for the Montgomery Ladder later on, since $P\odot_M^{Pro}Q$ will be a fixed, known point.

Based on this, we can define two pseudo-operations $xAdd$ and $xDbl$ which mostly parallel the Chord Rule and Tangent Rule of the group law $\odot_M^{Aff}$.
We define these operations analog to the approach presented in chapter 3 of \cite{costello_montgomery_2018}.

\begin{definition}[Pseudooperation $xAdd$]
    Let $M_{A,B}(\mathbb{F}P^2)$ be a Montgomery curve.
    Let $P, Q\in M_{A,B}(\mathbb{F}P^2)$ with $P\neq Q$.
    We define:
    \begin{align*}
        xAdd(x(Q), x(P), x(P\odot_M^{Pro}Q^{-1})) := x(P\odot_M^{Pro} Q)
    \end{align*}
    We can compute $xAdd$ as presented in \cref{alg:xAdd}.
    The corresponding circuit $\mathfrak{C}_{xAdd}$ is illustrated in \cref{fig:circuit:xAdd}.
\end{definition}

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Pseudooperation $xAdd$ used to compute $x(P\odot_M^{Pro} Q)$ from $x(P), x(Q)$ and $x(P\odot_M^{Pro}Q^{-1})$ with $P,Q\in M_{A,B}(\mathbb{F}P^2)$ and $P\neq Q$.}
    \label{alg:xAdd}
    \begin{algorithmic}
    \Procedure{$xAdd$}{$x(Q), x(P), x(P\odot_M^{Pro}Q^{-1})$}
        \State $[X_P:Z_P]:=x(P);\: [X_Q:Z_Q]:=x(Q);\: [X_{\odot^{-1}}, Z_{\odot^{-1}}]:=x(P\odot_m^{Pro}Q^{-1})$
        \State $X_\odot\gets  Z_{\odot^{-1}}((X_P-Z_P)(X_Q+Z_Q)+(X_P+Z_P)(X_Q-Z_Q))^2$
        \State $Z_\odot\gets  X_{\odot^{-1}}((X_P-Z_P)(X_Q+Z_Q)-(X_P+Z_P)(X_Q-Z_Q))^2$
        \State \Return $[X_\odot:Z_\odot]$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

\subsubsection{The Montgomery Ladder}
\label{section:ellipticCurves:montgomeryLadder}
The Montgomery Ladder is based on the Square and Multiply approach presented earlier in \cref{alg:fastExp}.
Analogously to Square and Multiply, we can compute exponentiations $P^k$ for some $P\in M_{A,B}(\mathbb{F}P^2)$ and $k\in\mathbb{N}$.
However, we can also employ $xAdd$ and $xDbl$ to more efficiently determine $x(P^k)$ based on $x(P)$. 
The Montgomery Ladder Algorithm computes $x(P^k)$ along with $x(P^{k+1})$ which we need later on to recover the $Y$ coordinate of $P^k$.

\begin{theorem}[Montgomery Ladder]
    Let $M_{A,B}(\mathbb{F}P^2)$ be a Montgomery curve.
    Let $P\in M_{A,B}(\mathbb{F}P^2)$.
    Let $k\in\mathbb{N}$, where $k=\sum\limits_{i=0}^{l-1}k_i2^i$ is a bit string of fixed length $l$.
    Then, we can compute $(x(P^k), x(P^{k+1}))$ with $Montgomery\text{-}Ladder$ defined in \cref{alg:montgomeryLadder}.
    The corresponding circuit $\mathfrak{C}_{Montgomery\text{-}Ladder}$ is illustrated in \cref{fig:circuit:montgomeryLadder}.
\end{theorem}

A proof of correctness of the $Montgomery\text{-}Ladder$ is given in section 4.1 of \cite{costello_montgomery_2018}.

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Montgomery Ladder algorithm used to compute $x(P^k)$ and $x(P^{k+1})$ from $x(P)$ with $P\in M_{A,B}(\mathbb{F}P^2)$ and $k\in\mathbb{N}$.
        Here, $k=\sum\limits_{i=0}^{l-1}k_i2^i$ is a bit string of fixed length $l$.
    }
    \label{alg:montgomeryLadder}
    \begin{algorithmic}
    \Procedure{$Montgomery\text{-}Ladder$}{$x(P), (k_0,\dots, k_{l-1})$}
        \State $(R_0, R_1)\gets (x([0:1:0]), x(P))$ 
        \State \Comment \parbox[t]{0.7\linewidth}{%
            Initialize $R_0$ with point at infinity and $R_1$ with $x(P)$. This is different compared to the standard version of the Montgomery Ladder.%
        }
        \For{$i=l-1$ \textbf{down to} $0$}
            \If{$k_i=0$}
                \State $(R_0, R_1)\gets (xDbl(R_0), xAdd(R_0, R_1, x(P)))$ 
                \State \Comment \parbox[t]{0.7\linewidth}{%
                    $R_1=x(P^{k+1})$ and $R_0=x(P^k)$ for some $k\in\mathbb{N}$ at all times.\\
                    Therefore, $x(P^{k+1}\odot_{M}^{Pro}(P^k)^{-1})=x(P)$.
                }
            \Else
                \State $(R_0, R_1)\gets (xAdd(R_0, R_1,
                 x(P)), xDbl(R_1))$
            \EndIf
        \EndFor
        \State \Return $(R_0, R_1)$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

    Here, $P\in M_{A,B}(\mathbb{F}P^2)$ and $(k_0,\dots, k_{l-1})$ is the binary representation of $k=\sum\limits_{i=0}^{l-1}k_i2^i$.

\newpage

Note that this is not the standard version of the Montgomery Ladder but slightly changed and originally intendended for constant-time execution \cite{costello_montgomery_2018}.
In the standard version of the Montgomery Ladder algorithm, we require the most significant bit of the exponent $k_{l-1}$ to be $1$. 
However, this means that the length of the bitstring representing $k$ changes based on the exponent $k$.
When we implement the Montgomery Ladder in a ciruit for our voting scheme this is problematic.
Since the exponent is not known during the compilation time of the circuit, we need to work with exponents of a fixed length $l$ where the most significatn bit might be zero.
Therefore, this version of the Montgomery Ladder is better suited for our purposes.

\subsubsection{$Y$-Recovery}
After performing the Montgomery Ladder algorithm on $x(P)$ and $n\in\mathbb{N}$ with $P\in M_{A,B}(\mathbb{F}P^2)$, we have $[X_k:Z_k]:=x(P^k)$ and $[X_{k+1}:Z_{k+1}]:=x(P^{k+1})$.
To reconstruct the $Y$ coordinate $Y_k$ of $P^{k}=[X_k:Y_k:Z_k]$, we use a recovery algorithm based on $P, x(P^k)$ and $x(P^{k+1})$ introduced by Okeya and Sakurai in \cite{goos_efficient_2001}.

\begin{theorem}[Okeya-Sakurai $Y$-Recovery]
    Let $M_{A,B}(\mathbb{F}P^2)$ be a Montgomery curve.
    Let $P=[X_P:Y_P:Z_P], Q, P\odot_{M}^{Pro}Q\in M_{A,B}(\mathbb{F}P^2)\setminus\{[0:1:0]\}$ with $Y_P\neq 0$.
    Then, we can reconstruct the complete $Q=Q_{rec}$, including the $Y$ coordinate from $(x_P, y_P):=I^{-1}(P)=\left(\frac{X_P}{Z_P}, \frac{Y_P}{Z_P}\right)$, $x(Q)=[X_Q:Z_Q]$ and $x(P\odot_M^{Pro} Q)=[X_\odot:Z_\odot]$ using \cref{alg:okSakYRecovery}.
    The corresponding circuit $\mathfrak{C}_{OkSak\text{-}Y\text{-}Recovery}$ is illustrated in \cref{fig:circuit:okSakYRecovery}.
\end{theorem}

The algorithm $OkSak\text{-}Y\text{-}Recovery$ we use here is based on Theorem 2 in \cite{goos_efficient_2001}. 
A proof of this theorem can be found in section B in the Appendix of \cite{goos_efficient_2001}.

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Okeya-Sakurai $Y$-Recovery used to recover $Q$ with $Y$ coordinate from $P$, $x(Q)$ and $x(P\odot_M^{Pro} Q)$.
    Here, $P=[X_P:Y_P:Z_P],P\odot_{M}^{Pro}Q\in M_{A,B}(\mathbb{F}P^2)\setminus\{[0:1:0]\}$, with $Y_P\neq 0$.}
    \label{alg:okSakYRecovery}
    \begin{algorithmic}
    \Procedure{$OkSak\text{-}Y\text{-}Recovery$}{$P, x(Q), x(P\odot_M^{Pro}Q)$}
        \State $[X_P:Y_P:Z_P]:=P;\: [X_Q:Z_Q]:=x(Q);\: [X_\odot:Z_\odot]:=x(P\odot_M^{Pro}Q)$
        \State $(x_P, y_P)\gets \left(\frac{X_P}{Z_P}, \frac{Y_P}{Z_P}\right)$
        \State $X_{Q_{rec}}\gets 2By_PZ_QZ_\odot X_Q$
        \State $Y_{Q_{rec}}\gets Z_\odot((X_Q+x_PZ_Q+2AZ_Q)(X_Qx_P+Z_Q)-2AZ_Q^2)-(X_Q-x_PZ_Q)^2X_\odot$
        \State $Z_{Q_{rec}}\gets 2By_PZ_Q^2Z_\odot$
        \State $Q_{rec} \gets  [X_{Q_{rec}}:Y_{Q_{rec}}:Z_{Q_{rec}}]$
        \State \Return $Q_{rec}$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

Now we can recover the $Y$ coordinate of a point $Q$ from its pseudopoint $x(Q)$, $P$ and $x(P\odot_M^{Pro} Q)$ if the preconditions for Okeya-Sakurai $Y$-Recovery are fulfilled.
However, we only want to employ the Okeya-Sakurai $Y$-Recovery for group exponentiations.
Here, we want to compute $P^k$ for some $P\in M_{A,B}^{Pro}$ and $k\in \mathbb{N}$.
The Montgomery Ladder produces the pseudopoints $x(P^k)$ and $x(P^{k+1})$.
In order to recover the correct $Y$-Coordinate of $P^k$, we can use the Okeya-Sakurai $Y$-Recovery with $P:=P$, $Q:=P^k$ and $P\odot_M^{Pro} Q:=P^{k+1}$ but need to treat some edge cases seperately:
\begin{itemize}
    \item First, we consider the case that $P=[0:1:0]$. 
    Here, we can conclude $P^{k+1}=P^k=[0:1:0]$.
    Note that the $OkSak\text{-}Y\text{-}Recovery$ returns the correct result $P^k=Q_{rec}=[X_{Q_{rec}}:Y_{Q_{rec}}:Z_{Q_{rec}}]=[0:Y_{Q_{rec}}:0]=[0:1:0]$ as long as $Y_{Q_{rec}}\neq 0$.
    We introduce a normalization function $Norm\text{-}Projective$ which covers the edge case $Y_{Q_{rec}}=0$ and does not change any other projective coordinates.
    \begin{align*}
        Norm\text{-}Projective:\{\mathbb{F}^3\}\to\{\mathbb{F}^3\}, [X:Y:Z]\to\begin{cases}
            \left[\frac{X}{Z}:\frac{Y}{Z}:1\right], &\:\text{if } Z\neq 0\\
            [0:1:0], &\:\text{otherwise}
        \end{cases}
    \end{align*}
    The circuit $\mathfrak{C}_{Norm\text{-}Projective}$ is visualized in \cref{fig:circuit:normProjective}.

    If we always perform $Norm\text{-}Projective$ on the output of $OkSak\text{-}Y\text{-}Recovery$, we don't need to check for $P^k=[0:1:0]$.
    \item Next, we know that $P^{k+1}=[0:1:0]\:\Leftrightarrow\:P^k=P^{-1}$.
    In this case, the correct output would be $P^k=Q_{rec}=P^{-1}=[X_P:-Y_P:Z_P]$.
\end{itemize}

We mostly follow the implementation in \cite{huber_hicolasnuberballotsnarks_2024} for the $Y$-Recovery on the output of the Montgomery Ladder.

\begin{theorem}[Montgomery Ladder $Y$-Recovery]
    Let $M_{A,B}(\mathbb{F}P^2)$ be a Montgomery curve.
    Let $P\in M_{A,B}(\mathbb{F}P^2)$, $k\in\mathbb{N}$.
    Then, we can reconstruct the complete $P^k=Q_{rec}$, including the $Y$ coordinate from $P, x(P^k)$ and $x(P^{k+1})$ using \cref{alg:montgomeryYRecovery}.
    The corresponding circuit $\mathfrak{C}_{Ladder\text{-}Y\text{-}Recovery}$ is illustrated in \cref{fig:circuit:ladderYRecovery}.
\end{theorem}

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{$Y$-Recovery used to recover $P^k$ with $Y$ coordinate from $P$, $x(P^k)$ and $x(P^{k+1})$ for some $k\in\mathbb{N}$.
    Here, $P\in M_{A,B}(\mathbb{F}P^2)$.}
    \label{alg:montgomeryYRecovery}
    \begin{algorithmic}
    \Procedure{$Ladder\text{-}Y\text{-}Recovery$}{$P, x(P^k), x(P^{k+1})$}
        \State $[X_P:Y_P:Z_P]:=P;\: [X_{P^k}:Z_{P^k}]:=x(P^k);\: [X_{P^{k+1}}:Z_{P^{k+1}}]:=x(P^{k+1})$
        \State $P_{rec}^k\gets [0:1:0]$
        \If{$Z_{P^{k+1}}=0$} \Comment \parbox[t]{0.7\linewidth}{%
                $\rightsquigarrow P^{k+1}=[0:1:0]$ and therefore $P^k=P^{-1}$.
            }
            \State $P_{rec}^k\gets [X_P:-Y_P:Z_P]$
        \Else
            \State $P_{rec}^k\gets OkSak\text{-}Y\text{-}Recovery(P, x(P^k), x(P^{k+1}))$
        \EndIf
        \State $P_{rec}^k\gets Norm\text{-}Projective(P_{rec}^k)$
        \State \Return $P_{rec}$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

\subsubsection{Full Exponentiation}
Now, we can combine the Montgomery Ladder with the $Y$-Recovery to compute full exponentiations $P^k$ for some $P\in M_{A,B}(\mathbb{F}P^2)$ and $k\in\mathbb{N}$.

\begin{definition}[Montgomery Exponentiation]
    Let $M_{A,B}(\mathbb{F}P^2)$ be a Montgomery curve.
    Let $P\in M_{A,B}(\mathbb{F}P^2)$.
    Let $k\in\mathbb{N}$, where $k=\sum\limits_{i=0}^{l-1}k_i2^i$ is a bit string of fixed length $l$.
    Then, we can compute $P^k=M\_Exp(P, (k_0, \dots, k_{l-1}))$ as described in \cref{alg:montgomeryExponentiation}.
    The corresponding circuit $\mathfrak{C}_{M_{A,B}(\mathbb{F}P^2)\_Exp}$ is illustrated in \cref{fig:circuit:montgomeryExp}.
\end{definition}

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Exponentiation on Montgomery curves. Used to compute $P^k$ where $P\in M_{A,B}^{Pro}$ and $k\in\mathbb{N}$. 
        Here, $k$ can be represented by a bit string of fixed length $l$.
    }
    \label{alg:montgomeryExponentiation}
    \begin{algorithmic}
    \Procedure{$M\_Exp$}{$P, k$}
        \State $(k_0,\dots, k_{l-1})\gets Num2Bits(k)$
        \State $(x(P^k), x(P^{k+1}))\gets  Montgomery\text{-}Ladder(x(P), (k_0,\dots,k_{l-1}))$
        \State $P^k\gets Ladder\text{-}Y\text{-}Recovery(P, x(P^k), x(P^{k+1}))$
        \State \Return $P^k$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

\section{Twisted Edwards Curves}
\label{section:ellipticCurves:twistedEdwards}

A third class of elliptic curves besides Short Weierstrass and Montgomery curves are Twisted Edwards curves.

As discussed in \cref{section:elGamalEncryption:precomputedPowers}, it can be benefitial to precompute powers of the public key elements $g$ and $h$ in \gls{eeg} if we want to encrypt many messages with the same public key.
This leads to \gls{ppeeg} which was introduced in \cref{def:ppeeg}.
If we use this approach, the montgomery ladder is no longer applicable.
The group law of Twisted Edwards curves is more efficient that that of Montgomery Curves.
Therefore, we can use this to compute group exponentiations more efficiently than with Montgomery Curves.
Then, \gls{ppeeg} based on Twisted Edwards curves is more efficient than \gls{eeg} based on Montgomery Curves.

\subsection{Twisted Edwards Form}
As with Montgomery Curves, we introduce Twisted Edwards curves similarly to \cite{least_authority_moonmath_2024}.

\begin{definition}[Twisted Edwards Curve]
    Let $(\mathbb{F},+,\cdot)$ be a finite field of chracteristic $p>3$. Let $a, d\in\mathbb{F}\setminus\{0\}$ with $a\neq d$.
    Then, the corresponding Twisted Edwards curve is the group $(TE_{a,d}(\mathbb{F}, \odot_{TE}))$, with:
    \begin{itemize}
        \item The set of points $TE_{a,d}(\mathbb{F})$:
        \begin{align*}
            TE_{a,d}(\mathbb{F})=\{(x,y)\in\mathbb{F}\times\mathbb{F}|a\cdot x^2+y^2=1+d\cdot x^2y^2\}
        \end{align*}
        \item The group law $\odot_{TE}$:\\
        Let $P_1, P_2\in TE_{a,d}(\mathbb{F})$ with $P_1=(x_1,y_1)$, $P_2=(x_2, y_2)$. Then:
        \begin{align*}
            P_\odot=(x_\odot, y_\odot)=(x_1, y_1)\odot_{TE} (x_2, y_2)=\left(\frac{x_1y_2+x_2y_1}{1+dx_1y_2y_1y_2},\frac{y_1y_2-ax_1x_2}{1-dx_1x_2y_1y_2}\right)
        \end{align*}
        We can also implement $\odot_{TE}$ inside a circuit $\mathfrak{C}_{TE_{a,d}(\mathbb{F})\_\odot}$ as shown in \cref{fig:circuit:teGroupLaw}.
    \end{itemize}
    The neutral element of $TE_{a,d}(\mathbb{F})$ is $(0,1)$. For $P=(x,y)\in TE_{a,d}(\mathbb{F})$, the inverse element is $P^{-1}:=(-x,y)$.
\end{definition}

\begin{figure}[h]
    \centering
    \includegraphics[scale=.7]{graphics/ellipticCurves/TEGroupLaw.pdf}
    \caption{Circuit $\mathfrak{C}_{TE_{a,d}(\mathbb{F})\_\odot}$ computing $P_1\odot_{TE}P_2$ for $P_1, P_2\in TE_{a,d}(\mathbb{F})$.
    }
    \label{fig:circuit:teGroupLaw}
\end{figure}

In constrast to Montgomery Curves there is no distinguished point at infinity $\mathcal{O}$ as the neutral element since the point $(0,1)$ is the neutral element of every Twisted Edwards curve.
Therefore, there is also no need for a projective version of Twisted Edwards curves.

\subsection{Group Exponentiation with Precomputed Powers}
We use a Twisted Edwards curve $TE_{a,d}(\mathbb{F})$ as the group for \gls{ppeeg} encryption because the Twisted Edwards group law is more efficient than the Montgomery group law.
The group exponentiation with precomputed powers of the element to exponentiate utilizes the approach presented in \cref{alg:fastExpPrecomputedPowers}.

\begin{definition}[Twisted Edwards Exponentiation with Precomputed Powers]
    Let $TE_{a,d}(\mathbb{F})$ be a Twisted Edwards curve. 
    Let $P=(x,y)\in TE_{a,d}(\mathbb{F})$ be a point, let $k\in\mathbb{N}$, where $k$ can be represented by a bit string of length $l$.
    Let $\left(P^{2^i}\right)_{i=0}^{l-1}$ be precomputed powers of $P$.
    Then, we can compute $P^k=TE\_PPFExp\left(\left(P^{2^i}\right)_{i=0}^{l-1}, k\right)$ as described in \cref{alg:twistedEdwardsExp}.
    
\end{definition}

The circuit $\mathfrak{C}_{TE_{a,d}(\mathbb{F})\_PPFExp}$ corresponding to $TE\_PPFExp$ is an instantiation of the circuit $\mathfrak{C}_{\mathbb{G}\_PPFExp}$ (\cref{fig:circuit:ppfexp}) where we choose $\mathbb{G}:=TE_{a,d}(\mathbb{G})$.
Note that this also requires the circuits $\mathfrak{C}_{Num2Bits}$ (\cref{fig:circuit:numBits}), $\mathfrak{C}_{TE_{a,d}(\mathbb{F})\_\odot}$ (\cref{fig:circuit:teGroupLaw}) and $\mathfrak{C}_{TE_{a,d}(\mathbb{F}\_If\text{-}Else)}$ as building blocks.
$\mathfrak{C}_{TE_{a,d}(\mathbb{F})\-If\text{-}Else}$ (\cref{fig:circuit:teIfElse}) is defined analogously to $\mathfrak{C}_{xIf\text{-}Else}$ (\cref{fig:circuit:xIfElse}).

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Exponentiation on Twisted Edwards curves. Used used to compute $P^k$ for $P\in TE_{a,d}(\mathbb{F})$ and $k\in\mathbb{N}$.
    Here, $k$ can be represented as a bitstring of length $l$ and $P^{2^i}\:\forall i\in\{0,\dots, l-1\}$ are precomputed powers of $P$.}
    \label{alg:twistedEdwardsExp}
    \begin{algorithmic}
    \Procedure{$TE\_PPFExp$}{$\left(P^{2^i}\right)_{i=0}^{l-1}, \left(k_i\right)_{i=0}^{l-1}$}
        \State $res\gets (0,1)$ \Comment Neutral element
        \For{$i=0$ \textbf{ to } $l-1$}
            \If{$k_i=1$}
                $res\gets P^{2^i}\odot_{TE} res$
            \EndIf
        \EndFor
        \State \Return $res$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

\subsection{Isomorphism with Montgomery Curves}
For every Twisted Edwards curve, there is an isomorphic Montgomery curve and vice versa.
Thus, we can transform between these two representations freely.
Additionally, it is typically quite simple to switch the two curves out for each other in most use cases such as our voting schemes.

\begin{theorem}[Twisted Edwards-Montgomery Isomorphism]
    Let $TE_{a,d}(\mathbb{F})$ be a Twisted Edwards curve.
    Then, for $A:=\frac{4}{a-d}$ and $B:=\frac{2(a+d)}{a-d}$ \cite{least_authority_moonmath_2024}, the Montgomery curve $M_{A,B}(\mathbb{F})$ is isomorphic to $TE_{a,d}(\mathbb{F})$.
    We can transform from the Twisted Edwards to the Montgomery curve with the function $TE\_to\_M:TE_{a,d}(\mathbb{F})\to M_{A,B}(\mathbb{F})$. 
    Let $P=(x,y)\in TE_{a,d}(\mathbb{F})$. Then \cite{costello_montgomery_2018}:
    \begin{align*}
        TE\_to\_M(P)=\begin{cases}
            \left(\frac{1+y}{1-y}, \frac{1+y}{(1-y)x}\right),\quad\text{if } (x,y)\neq (0,1)\\
            \mathcal{O},\quad\text{otherwise}
        \end{cases}
    \end{align*}

    Let $M_{A',B'}(\mathbb{F})$ be a Montgomery curve.
    Then, for $a':=\frac{A'+2}{B'}$ and $d':=\frac{A'-2}{B'}$ \cite{least_authority_moonmath_2024}, the Twisted Edwards curve $TE_{a',d'}(\mathbb{F})$ is isomorphic to $M_{A',B'}(\mathbb{F})$.
    We can transform from the Montgomers to the Twisted Edwards curve with the function $M\_to\_TE:M_{A',B'}(\mathbb{F})\to TE_{a',d'}(\mathbb{F})$. 
    Let $P'\in M_{A',B'}(\mathbb{F})$. Then \cite{costello_montgomery_2018}:
    \begin{align*}
        M\_to\_TE(P)=\begin{cases}
            \left(\frac{x'}{y'}, \frac{x'-1}{x'+1}\right),\quad\text{if } P'\neq\mathcal{O} \text{ with } P'=(x',y')\\
            (0,1),\quad\text{otherwise}
        \end{cases}
    \end{align*}
\end{theorem}
