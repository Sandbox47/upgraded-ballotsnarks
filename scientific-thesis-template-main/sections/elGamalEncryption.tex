\chapter{ElGamal Encryption}
\label{chap:elGamalEncryption}

We use \gls{eeg} as the additively homomorphic encrpytion scheme in our voting schemes.
To understand \gls{eeg} we first introduce ElGamal Encryption.

\section{ElGamal Encryption Scheme}
ElGamal Encryption is an asymmetric encryption scheme. 
Its security is based on the \gls{ddh} problem which is related to the Discrete Logarithm problem.

\begin{definition}[Discrete Logarithm Problem]
    Let $(\mathbb{G}, \odot_{\mathbb{G}})$ be a cyclic group of order $n:=|G|$ and $g$ be a generator of $\mathbb{G}$.
    Then, the discrete logarithm problem is computing the function $dlog_{(\mathbb{G},g)}:G\to\mathbb{Z}_n, dlog_{(\mathbb{G},g)}(g^{a})=a\:\forall a\in\mathbb{Z}_n$.\cref{footnote:groupExponentiation}
\end{definition}

\begin{definition}[\gls{ddh} Problem]
    Let $(\mathbb{G}, \odot_{\mathbb{G}})$ be a cyclic group of order $n:=|G|$ and let $g$ be a generator of $\mathbb{G}$.
    Let $\text{Un}$ be the probability distribution which outputs triples of the form $(g^{a}, g^{b}, g^{c})$ for $a, b, c$ chosen uniformly at random in $\{0, \dots, n-1\}$.\cref{footnote:groupExponentiation}
    Let $\text{DH}$ be the probability distribution which outputs triples of the form $(g^{a}, g^{b}, g^{ab})$ for $a, b$ chosen uniformly at random in $\{0, \dots, n-1\}$.
    Then, the \gls{ddh} problem is to distinguish $\text{Un}$ and $\text{DH}$.
\end{definition}

Formally, the ElGamal Encryption scheme is defined as follows.
We mostly follow the definition in construction 12.16 of \cite{katz_introduction_2021}.

\begin{definition}[ElGamal Encryption Scheme]
    \label{def:elGamalEncryption}
    The ElGamal asymmetric encryption scheme is $\mathcal{S}=(Gen(1^\eta), Enc, Dec)$. Here:
    \begin{itemize}
        \item $Gen(1^\eta)$: First, this algorithm generates a cyclic group $(\mathbb{G}, \odot_{\mathbb{G}})$ along with a generator $g$ of this group and computes the order of the group $n:=|\mathbb{G}|$.
        Then, it chooses an $x\in\mathbb{Z}_n$ uniformly at random, sets $h:=g^x$ and returns $(pk, sk)=((\mathbb{G}, n, g, h), (\mathbb{G}, n, g, x))$ as the public-private key pair.
        \footnote{\label{footnote:groupExponentiation} 
            We define $g^k := \underbrace{g\odot_{\mathbb{G}}\dots\odot_{\mathbb{G}}g}_{k \text{ times}}\quad\forall g\in\mathbb{G}\forall k\in\mathbb{N}$ with $g^0:=e_{\mathbb{G}}$. (Here, $e_{\mathbb{G}}$ denotes the neutral element of $\mathbb{G}$.)
        }
        \item $Enc(m, pk):$ To encrypt a message $m\in \mathbb{G}$ with $pk=(\mathbb{G}, n, g, h)$, the algorithm chooses $y\in\mathbb{Z}_n$ uniformly at random and outputs the ciphertext
        \begin{align*}
            c:=(g^y, h^y\odot_{\mathbb{G}} m)
        \end{align*}
        We also denote the encryption of $m$ under public key $pk$ and with a specified value $y$ by $Enc^{y}(m, pk)$.
        \item $Dec(c, sk):$ To decrypt a ciphertext $c=(c_0, c_1)$ with a secret key $sk=(\mathbb{G}, n, g, x)$, the algorithm computes the plain message as \begin{align*}
            m:=c_1\odot_{\mathbb{G}} (c_0^x)^{-1}
        \end{align*}
    \end{itemize}
\end{definition}

We can show that ElGamal Encryption is secure as long as the \gls{ddh} problem is computationally hard to solve for the groups $\mathbb{G}$ generated by $Gen(1^\eta)$.
For a formal definition of the secuity of ElGamal and security proof, please refer to Theorem 12.18 of \cite{katz_introduction_2021}.
Note that computational hardness of the \gls{ddh} problem implies computational hardness of the Discrete Logarithm problem but not necessarily vice versa \cite{least_authority_moonmath_2024}.

\section{Homomorphic Encryption Schemes}
Homomorphic encryption schemes allow us to obtain the encrypted result of some computation on the plaintexts by some (other) computation on the ciphertext.
For instance, a homomorphic encryption scheme could allow aggregation of ballots by some computation on the corresponding encrypted ballots. 
This is exactly, what we require in voting schemes.

We define homomorphic encryption schemes as presented in definition 15.14 of \cite{katz_introduction_2021} and just change the notation slightly to match our encryption scheme notation.

\begin{definition}[Homomorphic Encryption Scheme]
    Let $\mathcal{S}=(Gen, Enc, Dec)$ be an encryption scheme. We call $\mathcal{S}$ homomorphic if for all $(pk, sk)$ generated by $Gen(1^\eta)$, it is possible to define groups $\mathcal{M}$ and $\mathcal{C}$ (depending on $pk$ only) such that:
    \begin{itemize}
        \item The message space is $\mathcal{M}$, and all ciphertexts output by $Enc(\cdot, pk)$ are elements of $\mathcal{C}$. 
        We write $\mathcal{M}$ as a group with operation $\odot_{\mathcal{M}}$ and $\mathcal{C}$ as a group with operation $\odot_{\mathcal{C}}$.
        \item For any $m_1, m_2\in \mathcal{M}$, any $c_1$ output by $Enc(m_1, pk)$ and any $c_2$ output by $Enc(m_2, pk)$, it holds that \begin{align*}
            Dec(c_1\odot_{\mathcal{C}} c_2, sk) = m_1 \odot_{\mathcal{M}} m_2 \text{.}
        \end{align*}
        %Note that this is equivalent to $c_1\odot_{\mathcal{C}} c_2$ being a possible encryption of $m_1 \odot_{\mathcal{M}} m_2$.
        Moreover, the distribution on ciphertexts obtained by encrypting $m_1$, encrypting $m_2$, and then applying $\odot_{\mathcal{C}}$ to these ciphertexts is identical to the distribution on ciphertexts obtained by encrypting $m_1\odot_{\mathcal{M}}m_2$.
    \end{itemize}
\end{definition}

With this definition, we can see that ElGamal encryption is homomorphic.

Let $(pk, sk)=((\mathbb{G}, n, g, x), (\mathbb{G}, n, g, h))$ be a public key generated by $Gen(1^\eta)$ in the ElGamal encryption scheme $\mathcal{S}$. 
As previously, $\odot_{\mathbb{G}}$ denotes the operation of $\mathbb{G}$.
We choose $\mathcal{M}=\mathbb{G}$ and $\mathcal{C}=\mathbb{G}\times \mathbb{G}$.
Let $m_1, m_2\in\mathbb{G}$ be two messages and let $c_1, c_2$ be the corresponding ciphertexts.

We define:
\begin{itemize}
    \item For $\mathcal{M}$:
    \begin{align*}
        m_1\odot_{\mathcal{M}}m_2 := m_1 \odot_{\mathbb{G}} m_2
    \end{align*}
    \item For $\mathcal{C}$:
    \begin{align*}
        c_1\odot_{\mathcal{C}}c_2 &= (c_{1,0}, c_{1,1})\odot_\mathcal{C} (c_{2,0}, c_{2,1})\\
        &= (c_{1,0}\odot_\mathbb{G} c_{2,0}, c_{1,1}\odot_\mathbb{G} c_{2,1})
        %c_1\odot_{\mathcal{C}}c_2 &= Enc(m_1, pk)\odot_{\mathcal{C}} Enc(m_2, pk)\\
        %&= (g^{y_1}, h^{y_1}\odot_{\mathbb{G}} m_1)\odot_{\mathcal{C}} (g^{y_2}, h^{y_2}\odot_{\mathbb{G}} m_2)\\
        %:&= (g^{y_1+y_2}, h^{y_1+y_2}\odot_{\mathbb{G}} m_1\odot_{\mathbb{G}} m_2)
    \end{align*}
\end{itemize}

Then, we get:
\begin{align*}
    Dec(c_1 \odot_{\mathcal{C}} c_2, sk) & = Dec(Enc(m_1, pk)\odot_{\mathcal{C}} Enc(m_2, pk), sk)\\
    & = Dec(Enc(m_1, (\mathbb{G}, n, g, h))\odot_{\mathcal{C}} Enc(m_2, pk), (\mathbb{G}, n, g ,h))\\
    & = Dec((g^{y_1}, h^{y_1}\odot_{\mathbb{G}}m_1)\odot_{\mathcal{C}} (g^{y_2}, h^{y_2}\odot_{\mathbb{G}}m_2), sk)\\
    & = Dec((g^{y_1+y_2}, h^{y_1+y_2}\odot_{\mathbb{G}}m_1\odot_{\mathbb{G}} m_2), (\mathbb{G}, n, g, x))\\
    & = h^{y_1+y_2}m_1\odot_{\mathbb{G}} m_2 \odot_{\mathbb{G}} ((g^{y_1+y_2})^x)^{-1}\\
    & = (g^x)^{y_1+y_2}\odot_{\mathbb{G}} ((g^{y_1+y_2})^x)^{-1}\odot_{\mathbb{G}} m_1\odot_{\mathbb{G}} m_2\\
    & = m_1\odot_{\mathbb{G}} m_2\\
    & = m_1\odot_{\mathcal{M}}m_2
\end{align*}

Despite the homomorphic property, standard ElGamal is not useful when it comes to aggregating ballots.
To aggregate two ballots $b_1, b_2$ from a choice space $C$, we need to add them by performing the standard (component-wise) addition $b_1+b_2$.
However, ElGamal encryption would only allow us to compute the (component-wise) operation $b_1\odot_{\mathbb{G}} b_2$ of two ballots $b_1$ and $b_2$ by performing the operation $\mathcal{C}$ on the corresponding ciphertexts.
The operation $\odot_{\mathbb{G}}$ depends on the generated group $\mathbb{G}$ and in most cases is closer to the standard multiplication \glqq$\cdot$\grqq\ than the addition \glqq$+$\grqq.
To allow for (component-wise) addition of ballots, we need a slight variation of ElGamal, namely Exponential ElGamal.

\section{Exponential ElGamal Encryption Scheme}
\gls{eeg} encryption is very similar to normal ElGamal encryption.
The only difference is, that instead of encrypting a message $m$ with a public key $(\mathbb{G}, n, g, h)$ directly, we encrypt $g^m$.
Therefore, $m$ must now be chosen from $\{0,\dots, n-1\}$ instead of from $\mathbb{G}$.
\gls{eeg} encryption is benefitial as it is additively homomorphic independent of the chosen group $\mathbb{G}$ and therefore allows easy aggregation of encrypted ballots in our election scheme.

Formally, we define \gls{eeg} encryption as in \cite{aldini_cryptographic_2014} but align the notation with our notation of encryption schemes.

\begin{definition}[Exponential ElGamal Encryption Scheme]
    \label{def:eeg}
    The \gls{eeg} encryption scheme is $EEG(\mathbb{G})=(Gen(1^\eta), Enc, Dec)$. 
    Here:
    \begin{itemize}
        \item $Gen(1^\eta)$ is defined as in \cref{def:elGamalEncryption} but we fix the generated cyclic group to be $\mathbb{G}$.\footnote{
            \label{footnote:fixedGeneratedGroup}
            Fixing the group over which \gls{eeg} operates is useful for our voting schemes since we implement encryption over a fixed elliptic curve for efficiency and circuit compatibility reasons.
        }
        \item $Enc(m, pk):$ To encrypt a message $m\in \mathbb{Z}_n$ with $pk=(\mathbb{G}, n, g, h)$, the algorithm chooses $y\in\mathbb{Z}_n$ uniformly at random and outputs the ciphertext
        \begin{align*}
            c:=(g^y, h^y\odot_{\mathbb{G}} g^m)
        \end{align*}
        We also denote the encryption of $m$ under public key $pk$ and with a specified value $y$ by $Enc^{y}(m, pk)$.
        \item $Dec(c, sk):$ To decrypt a ciphertext $c=(c_0, c_1)$ with a secret key $sk=(\mathbb{G}, n, g, x)$, the algorithm first computes $\tilde{m}=g^m$ for the original message $m$: \begin{align*}
            \tilde{m}:=c_1\odot_{\mathbb{G}} (c_0^x)^{-1}
        \end{align*}
        To compute the original message $m$ the algorithm computes $m$ such that $g^m=\tilde{m}$.\footnote{\label{footnote:eegDiscreteLogProblem}
            Note that this is equivalent to the Discrete Logarithm problem and assumed to be computationally hard for the groups $\mathbb{G}$ chosen in a secure \gls{eeg} encryption scheme if $m$ can be any element of $\mathbb{G}$.

            However, in practical applications such as voting schemes the message is typically one entry in a tally of ballots, where thi individual ballots are chosen from a choice space $C$.
            We denote the set of allowed values for the entries in a ballot from $C$ by $C_{entry}$, which is typically very small. 
            In an election with $n$ voters, the set of possible values for an entry in the plain tally $T_{entry}$ is therefore at most $n$ times larger than $C_{entry}$ and thus typically still quite small.
            Then computing $m$ is possible by a guess-and-check approach where we try all elements $m\in T_{entry}$ and return the current choice if $g^m=\tilde{m}$.
            
            In our example voting scheme from \cref{chap:votingschemes}, we had $n=5$ voters and $C_{entry}=\{0,1\}$.
            By aggregating the individual entries of ballots, in the plain tally, we get entries $m$ in the set $T_{entry}=\{0,1,2,3,4,5\}$.
        }
    \end{itemize}
\end{definition}

We can conclude that \gls{eeg} encryption is homomorphic.

Let $(pk, sk)=((\mathbb{G}, n, g, x), (\mathbb{G}, n, g, h))$ be a public key generated by $Gen(1^\eta)$ in the \gls{eeg} encryption scheme $\mathcal{S}$. 
As previously, $\odot_{\mathbb{G}}$ denotes the operation of $\mathbb{G}$.
We choose $\mathcal{M}=\{0,\dots, n-1\}$ and $\mathcal{C}=\mathbb{G}\times \mathbb{G}$.
Let $m_1, m_2\in\mathcal{M}$ be two messages and let $c_1, c_2$ be the corresponding ciphertexts.

We define:
\begin{itemize}
    \item For $\mathcal{M}$:
    \begin{align*}
        m_1\odot_{\mathcal{M}}m_2 := m_1 + m_2
    \end{align*}
    Here, \glqq$+$\grqq\ denotes the standard addition in $\mathbb{N}$.
    \item For $\mathcal{C}$:
    \begin{align*}
        c_1\odot_{\mathcal{C}}c_2 &= Enc(m_1, pk)\odot_{\mathcal{C}} Enc(m_2, pk)\\
        &= (g^{y_1}, h^{y_1}\odot_{\mathbb{G}} g^{m_1})\odot_{\mathcal{C}} (g^{y_2}, h^{y_2}\odot_{\mathbb{G}} g^{m_2})\\
        :&= (g^{y_1+y_2}, h^{y_1+y_2}\odot_{\mathbb{G}} g^{m_1+m_2})
    \end{align*}
\end{itemize}

Then, we get:
\begin{align*}
    Dec(c_1 \odot_{\mathcal{C}} c_2, sk) & = Dec(Enc(m_1, pk)\odot_{\mathcal{C}} Enc(m_2, pk), sk)\\
    & = Dec(Enc(m_1, (\mathbb{G}, n, g, h))\odot_{\mathcal{C}} Enc(m_2, pk), (\mathbb{G}, n, g ,h))\\
    & = Dec((g^{y_1}, h^{y_1}\odot_{\mathbb{G}}g^{m_1})\odot_{\mathcal{C}} (g^{y_2}, h^{y_2}\odot_{\mathbb{G}}g^{m_2}), sk)\\
    & = Dec((g^{y_1+y_2}, h^{y_1+y_2}\odot_{\mathbb{G}}g^{m_1+m_2}), (\mathbb{G}, n, g, x))\\
    & = dlog_{(\mathbb{G},g)}(h^{y_1+y_2}\odot_{\mathbb{G}}g^{m_1+m_2}\odot_{\mathbb{G}} ((g^{y_1+y_2})^x)^{-1})\\
    & = dlog_{(\mathbb{G},g)}((g^x)^{y_1+y_2}\odot_{\mathbb{G}} ((g^{y_1+y_2})^x)^{-1}\odot_{\mathbb{G}}g^{m_1+m_2})\\
    & = dlog_{(\mathbb{G},g)}(g^{m_1+m_2})\\
    & = m_1+m_2\\
    & = m_1\odot_{\mathcal{M}}m_2
\end{align*}

As we can see, \gls{eeg} encryption yields additive homomorphism as needed for voting schemes.
Let $(pk, sk)=((\mathbb{G}, n, g, x), (\mathbb{G}, n, g, h))$ be a public key generated by $Gen(1^\eta)$ in the \gls{eeg} encryption scheme $\mathcal{S}$ as before.
Assume, we want to compute an encrypted total of two ballots $b_1, B_2\in C\subsetneq\{0,\dots, n-1\}$.
We can compute a possible encryption of the total $b_1+b_2$ as follows.\footnote{\label{footnote:probEncEEG}
    $Enc$ in \gls{eeg} is a probabilistic algorithm.
    Therefore, there are multiple possible encryptions of $b_1 + b_2$.
    However, all of the encryptions decrpyt to the correct tally $b_1+b_2$.
}
\begin{align*}
    Enc(b_1+b_2, pk) &= Enc(b_1, pk) \odot_{\mathcal{C}} Enc(b_1, pk)
\end{align*}
In a voting scheme, the $VA$ can aggregate all submitted ballots in this way and decrypt just the final tally at the end.

In order to create a \gls{zkp} of correct encryption of a ballot in our voting scheme, we need to implement the encryption function $Enc$ of the used encryption scheme inside a circuit.
This also requires circuits $\mathfrak{C}_{\mathbb{G}\_\odot}$ and $\mathfrak{C}_{\mathbb{G}\_Exp}$ for computing group exponentiation $g^k$ and the group law $g_1\odot_{\mathbb{G}} g_2$.
Since these circuits depend on the concrete choice of $\mathbb{G}$, we will use them as black box circuits as shown in \cref{fig:circuit:groupLaw} and \cref{fig:circuit:groupExp} for now.

\section{Precomputed Powers Exponential ElGamal Encryption Scheme}
\label{section:elGamalEncryption:precomputedPowers}
The overwhelming majority of the computational effort needed to compute \gls{eeg} is caused by the exponentiation of group elements.
Let $\mathcal{S}=(Gen(1^\eta), Enc, Dec)$ be an \gls{eeg} encryption scheme and $pk=(\mathbb{G}, n, g, h)$ a public key in this scheme.
Then, to compute the encryption of a message $m\in\{0, \dots, n-1\}$ as introduced in \cref{def:eeg}, we need to compute three group exponentiations $g^y, h^y$ and $g^m$ where $y\probgets \{0,\dots, n-1\}$.

Let $\mathbb{G}$ be a group, let $g\in\mathbb{G}$ and let $k\in\{0,\dots, |\mathbb{G}|-1\}$.
Then, the naive way of computing $g^k$ as $g^k=\underbrace{g\odot_{\mathbb{G}}\cdots\odot_\mathbb{G} g}_{k \text{ times}}$ requires $k-1$ applications of the group law.
In our use case, $\mathbb{G}$ has an order that can be expressed as a $251$-bit number. 
Therefore, $k$ can also be a number with up to $251$ bits.
Then, computing $g^k$ naively is completely infeasible.
A more efficient approach is the Square and Multiply method as introduced in chapter 9 of \cite{cohen_handbook_2005} which needs $\mathcal{O}(\log(k))$ applications of the group law.
Here, we use the binary representation $(k_0,\dots, k_{l-1})$ of $k=\sum\limits_{i=0}^{l-1}k_i\cdot 2^{i}$ to compute $g^k$:
\begin{align*}
    g^k&=g^{\left(\sum\limits_{i=0}^{l-1}k_i\cdot 2^{i}\right)} = \prod_{i=0}^{l-1} g^{k_i\cdot 2^{i}} = \prod_{i=0}^{l-2} \left(g^{2^{i}}\right)^{k_i}
\end{align*}
This is benefitial, since we can compute $g^{2^i}$ by iterated squaring of $g$.

The algorithm $FastExp(g, k)$ to compute $g^k$ in $\mathbb{G}$ with the Square and Multiply approach is presented in \cref{alg:fastExp}.
Here, we use the function $Num2Bits(k)$ to compute the \gls{lsb} binary representation $(k_0,\dots,k_{l-1})$ of $k$, with $k=\sum\limits_{i=0}^{l-1}k_i2^i$.

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Algorithm $FastExp$ used to compute $g^k$ for $g\in\mathbb{G}$ and $k\in\{0,\dots, |\mathbb{G}|-1\}$.
    Here, $k$ can be represented as bitstring of length $l$.}
    \label{alg:fastExp}
    \begin{algorithmic}
    \Procedure{$FastExp$}{$g, k$}
        \State $y\gets e_\mathbb{G}$ \Comment $e_\mathbb{G}$ is the neutral element of $\mathbb{G}$
        \State $(k_0, \dots, k_{l-1})\gets Num2Bits(k)$
        \For{$i=l-1$ \textbf{ down to } $0$}
            \State $y\gets y^2=y\odot_\mathbb{G} y$
            \If{$k_i=1$}
                \State $y\gets g\odot_\mathbb{G} y$
            \EndIf
        \EndFor
        \State \Return $y$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

This is the basis for the Montgomery Ladder Algorithm we use in \cref{section:ellipticCurves:montgomery} to compute group exponentiations for montgomery curves.

It is possible to optimize computing group exponentiation further if we provide $g^{2^i}\:\forall i\in\{0,\dots, l-1\}$ to the exponentiation algorithm.
This approach, using precomputed powers of the element to compute $g^k$, is defined in \cref{alg:fastExpPrecomputedPowers}.

\begin{Algorithmus}[h] %Use the environment only if you want to place the algorithm similar to graphics from TeX
    \caption{Algorithm $PrecomputedPowersFastExp$ used to compute $g^k$ for $g\in\mathbb{G}$ and $k\in\{0,\dots, |\mathbb{G}|-1\}$.
    Here, $k$ can be represented as bitstring of length $l$ and $g^{2^i}\:\forall i\in\{0,\dots, l-1\}$ are precomputed powers of $g$.}
    \label{alg:fastExpPrecomputedPowers}
    \begin{algorithmic}
    \Procedure{$PrecomputedPowersFastExp$}{$(g^{2^0}, \dots, g^{2^{l-1}}), k$}
        \State $y\gets e_\mathbb{G}$ \Comment $e_\mathbb{G}$ is the neutral element of $\mathbb{G}$
        \State $(k_0, \dots, k_{l-1})\gets Num2Bits(k)$ \Comment \gls{lsb} binary representation of $k$
        \For{$i=0$ \textbf{ to } $l-1$}
            \If{$k_i=1$}
                \State $y\gets g^{2^i}\odot_\mathbb{G} y$
            \EndIf
        \EndFor
        \State \Return $y$
    \EndProcedure
    \end{algorithmic}
\end{Algorithmus}

To represent $PrecomputedPowersFastExp$ as a circuit, we need additional circuits $\mathfrak{C}_{Num2Bits}$ (\cref{fig:circuit:numBits}) to compute $Num2Bits(k)$ and $\mathfrak{C}_{\mathbb{G}\_If\text{-}Else}$ to enable branching.
%$Num2Bits$ is a standard template and preimplemented in the circuit language Circom we use here \cite{circomlib}.
$\mathfrak{C}_{\mathbb{G}\_If\text{-}Else}$ depends on the choice of $\mathbb{G}$.
Therefore, we will use it as a black box circuit as shown in \cref{fig:circuit:groupIfElse} for now.
Together with the black box circuit $\mathfrak{C}_{\mathbb{G}\_\odot}$ from \cref{fig:circuit:groupLaw}, we can then implement $PrecomputedPowersFastExp$ as the circuit $\mathfrak{C}_{\mathbb{G}\_PPFExp}$ presented in \cref{fig:circuit:ppfexp}.

The algorithm $PrecomputedPowersFastExp$ is used in \cref{section:ellipticCurves:twistedEdwards} to compute group exponentiation for Twisted Edwards curves.
Precomputing $g^{2^i}\:\forall i\in\{0,\dots, l-1\}$ can be adventageus, when computing multiple group exponentiations with exponents that have at most $l$ bits.
In this case, we only need to compute the powers of $g$ once and can reuse them for all exponentiations.
One such use case is \gls{eeg} encryption.
In our voting schemes, we want to encrypt ballots $b$ with entries $b_0,\dots, b_{m-1}$.
With \gls{eeg} encryption, we reuse the same public key $pk=(\mathbb{G}, n, g, h)$ to compute all entries $b'_0,\dots,b'_{m-1}$ of the encrypted ballot $b'=\mathcal{S}.Enc(b, pk)$ belonging to ballot $b$:
\begin{align*}
    \forall i\in\{0,\dots, m-1\}:\:b'_i=\mathcal{S}.Enc(b_i, pk)=(g^{y_i}, h^{y_i}\odot_\mathbb{G} g^{b_i})\text{ with } y_i\probgets \mathbb{Z}_n
\end{align*}
Additionally, we reuse the key $pk$ to compute encryptions of other ballots.
With the current definition of \gls{eeg}, we can use $FastExp$ to compute $g^{y_i}, h^{y_i}$ and $g^{b_i}$.
However, we can only use $PrecomputedPowersFastExp$ for that purpose if we compute $g^{2^0},\dots, g^{2^{\lfloor\log_2(n)\rfloor}}$ and $h^{2^0},\dots, h^{2^{\lfloor\log_2(n)\rfloor}}$ before.
This requires the same computational effort as $FastExp$.

Therefore, we propose a slightly modified version of \gls{eeg} which includes the precomputed powers of $g$ and $h$ in the public key $pk$: \gls{ppeeg}.

\begin{definition}[Precomputed Powers Exponential ElGamal Encryption]
    \label{def:ppeeg}
    The \gls{ppeeg} encryption scheme is $PPEEG(\mathbb{G})=(Gen(1^\eta), Enc, Dec)$. 
    Here:
    \begin{itemize}
        \item $Gen(1^\eta)$: The algorithm generates $(\mathbb{G}, \odot_{\mathbb{G}}), g, x$ and $h$ as in \cref{def:elGamalEncryption} but with $\mathbb{G}$ being fixed as in \cref{def:eeg}.
        Then, it computes $g^{2^i}, h^{2^i}\:\forall i\in\{0,\dots, l-1\}$ with $l:=\lfloor\log_2(n)\rfloor+1$ and returns 
        \begin{align*}
            (pk, sk)=\left(\left(\mathbb{G}, n, \left(g^{2^i}\right)_{i=0}^{l}, \left(h^{2^i}\right)_{i=0}^{l}\right), (\mathbb{G}, n, g, x)\right) 
        \end{align*}
        as the public-private key pair.
        \item $Enc(m, pk)$ is defined as in \cref{def:eeg} but uses $PrecomputedPowersFastExp$ to compute the group exponentiations. 
        Specifically, for $y=\sum\limits_{i=0}^{l}y_i\cdot 2^i$ and $m=\sum\limits_{i=0}^{l}m_i\cdot 2^i$ with $l:=\lfloor\log_2(n)\rfloor+1$:
        \begin{align*}
            g^{y}&:= PrecomputedPowersFastExp\left(\left(g^{2^i}\right)_{i=0}^{l}, \left(y_i\right)_{i=0}^{l}\right)\\
            h^{y}&:= PrecomputedPowersFastExp\left(\left(h^{2^i}\right)_{i=0}^{l}, \left(y_i\right)_{i=0}^{l}\right)\\
            g^{m}&:= PrecomputedPowersFastExp\left(\left(g^{2^i}\right)_{i=0}^{l}, \left(m_i\right)_{i=0}^{l}\right)
        \end{align*}
        \item $Dec(c, sk)$ is defined as in \cref{def:eeg}.
    \end{itemize}
\end{definition}

Note that a \gls{ppeeg} encryption scheme $\mathcal{S}'$ is secure if the corresponding \gls{eeg} encryption scheme $\mathcal{S}$ is secure.
If an adversary $A'$ can break security in $\mathcal{S}'$, then we can construct an adversary $A$ on $\mathcal{S}$ which computes the powers $g^{2^i}, h^{2^i}\:\forall i\in\{0,\dots, \lfloor\log_2(n)\rfloor\}$ from the public key $pk$ and simulates $A'$ after that.
Then, $A$ will break security of $\mathcal{S}$.

Just as with \gls{eeg}, we can also compute encryption in a \gls{ppeeg} encryption scheme inside a circuit.
By providing the randomness $y$ used by $Enc$ as an additional input, we can implement $PPEEG(\mathbb{G}.Enc)$ as the $\mathfrak{C}_{PPEEG(\mathbb{G}.Enc)}$ (\cref{fig:circuit:ppeegEnc}).
